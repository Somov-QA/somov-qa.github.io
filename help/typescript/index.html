🔷 Основы TypeScript

Что такое TypeScript?

TypeScript — надмножество JavaScript, добавляющее статическую типизацию. Код компилируется в чистый JS.

Установка и настройка

# Установка TypeScript
npm install -g typescript

# Создание конфигурационного файла
tsc --init

# Компиляция файла
tsc app.ts

# Компиляция с наблюдением
tsc --watch

Базовый синтаксис

let message: string = "Привет, TypeScript!";

function sum(a: number, b: number): number {
    return a + b;
}

const result = sum(5, 10);

🏛️ Классы

class Person {
    name: string;
    private age: number;
    protected email: string;
    readonly id: number;
    constructor(name: string, age: number, email: string, id: number) {
        this.name = name; this.age = age; this.email = email; this.id = id;
    }
    greet(): string { return `Привет, ${this.name}!`; }
    get personAge(): number { return this.age; }
    set personAge(age: number) { if (age > 0 && age < 120) this.age = age; }
}

⚙️ Дженерики

function identity<T>(arg: T): T { return arg; }
let output = identity<string>("myString");

class Box<T> {
    private value: T;
    constructor(value: T) { this.value = value; }
    getValue(): T { return this.value; }
}

interface Lengthwise { length: number; }
function loggingIdentity<T extends Lengthwise>(arg: T): T {
    console.log(arg.length);
    return arg;
}

function merge<T extends object, U extends object>(obj1: T, obj2: U): T & U {
    return { ...obj1, ...obj2 };
}

function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
    return obj[key];
}

🚀 Продвинутые типы

Union / Intersection

function padLeft(value: string, padding: string | number): string { /* ... */ }
type BusinessDeveloper = BusinessPerson & Developer;

Type Guards

function isString(value: any): value is string { return typeof value === "string"; }
function process(value: string | number): string {
    if (isString(value)) return value.toUpperCase();
    return value.toString();
}

Literal Types & Discriminated Unions

type Direction = "up" | "down";
type DiceRoll = 1 | 2 | 3 | 4 | 5 | 6;

interface Square { kind: "square"; size: number; }
interface Circle { kind: "circle"; radius: number; }
type Shape = Square | Circle;
function area(shape: Shape): number {
    switch (shape.kind) {
        case "square": return shape.size ** 2;
        case "circle": return Math.PI * shape.radius ** 2;
    }
}

Mapped & Conditional Types

type Partial<T> = { [P in keyof T]?: T[P]; };
type Readonly<T> = { readonly [P in keyof T]: T[P]; };
type IsString<T> = T extends string ? true : false;
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

Utility Types

interface Todo { title: string; description: string; completed: boolean; }
type TodoPreview = Pick<Todo, "title" | "completed">;
type TodoWithoutDesc = Omit<Todo, "description">;
type Page = "home" | "about";
const nav: Record<Page, { title: string }> = { home: { title: "Главная" }, about: { title: "О нас" } };

✨ Декораторы

// Декоратор класса
function sealed(constructor: Function) { Object.seal(constructor); }

class Greeter { greeting: string; }

// Декоратор метода
function enumerable(value: boolean) {
    return function(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        descriptor.enumerable = value;
    };
}
class Example { @enumerable(false) greet() {} }

// Декоратор параметра

📚 Работа с библиотеками и типами

Установка определений типов

npm install @types/lodash --save-dev
import * as _ from 'lodash';

Создание своих объявлений (.d.ts)

// my-module.d.ts
declare module 'my-module' {
    export function myFunction(param: string): number;
    export class MyClass { constructor(name: string); getName(): string; }
}

Глобальные типы / Ambient декларации

// global.d.ts
declare global {
    interface Window { myGlobalFunction(): void; }
}
// Использование
window.myGlobalFunction();

declare const $: { (selector: string): any; ajax(url: string): void; };

Совет Всегда используйте @types для популярных библиотек или создавайте собственные декларации для чистоты кода.

🧠 Полезные приёмы

Keyof и lookup types

interface Person { name: string; age: number; }
type PersonKeys = keyof Person; // "name" | "age"

function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] { return obj[key]; }

Преобразование типов с помощью as const

const colors = ["red", "green", "blue"] as const;
type Color = typeof colors[number]; // "red" | "green" | "blue"

TypeScript расширяет JS мощной системой типов. Используйте эту шпаргалку для повседневной разработки, комбинируйте utility-типы и наслаждайтесь безопасностью кода.
