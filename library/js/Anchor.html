DOM Anchor

Свойства:

accessKey - Устанавливает или возвращает клавишу клавиатуры, чтобы получить доступ к ссылке

charset - Устанавливает или возвращает набор символов связанного ресурса

coords - Устанавливает или возвращает разделенный запятыми список с координатами, определяя активную область геометрии

href - Устанавливает или возвращает URL связанного ресурса

hreflang - Устанавливает или возвращает код языка связанного ресурса

id - Устанавливает или возвращает идентификатор ссылки

innerHTML - Устанавливает или возвращает текст ссылки

name - Задает или возвращает имя ссылки

rel - Устанавливает или возвращает отношение между текущим документом и целевым URL

rev - Устанавливает или возвращает отношение между текущим документом и целевым URL

shape - Устанавливает или возвращает активную область. Координаты задаются координатам

tabIndex - Устанавливает или возвращает индекс, который определяет порядок обхода для канала

target - Устанавливает или возвращает где открыть целевой URL

type - Устанавливает или возвращает тип MIME связанного ресурса


Методы:

blur() - Удаляет фокус из ссылки

focus() - Передает фокус на ссылку


События:

onBlur - Выполняет некоторый код, когда ссылка теряет фокус

onFocus - Выполняет некоторый код, когда ссылка получает фокус

<!DOCTYPE html>
<html>
<head>
   <title></title>

   <script type="text/javascript">
               function access()
               {
                       document.getElementById('myAnchor').accessKey="a" // горячие клавиши alt + a
               }

               function changeLink()
               {
                       document.getElementById('myLink').innerHTML="W3Schools"
                       document.getElementById('myLink').href="http://www.w3schools.com"
                       document.getElementById('myLink').target="_blank"
                       console.log("charset:", document.getElementById('myLink').charset);
                       console.log("coords:", document.getElementById('myLink').coords);
                       console.log("hreflang:", document.getElementById('myLink').hreflang);
                       console.log("id:", document.getElementById('myLink').id);
                       console.log("name:", document.getElementById('myLink').name);
                       console.log("rel:", document.getElementById('myLink').rel);
                       console.log("rev:", document.getElementById('myLink').rev);
                       console.log("shape:", document.getElementById('myLink').shape);
                       console.log("tabIndex:", document.getElementById('myLink').tabIndex);
                       console.log("type:", document.getElementById('myLink').type);
               }

               function getFocus()
               {
                       document.getElementById('myAnchor').focus()
               }

               function loseFocus()
               {
                       document.getElementById('myAnchor').blur()
               }

               function onFocus()
               {
                       console.log("onFocus: Get focus")
               }

               function onBlur()
               {
                       console.log("onBlur: Lose focus")
               }
               
       </script>

</head>
<body onload="access()">
<p>
   <a id="myAnchor" href="http://www.w3schools.com" onfocus="onFocus()" onblur="onBlur()">Visit W3Schools</a>
   <br /><br />
   <input type="button" onclick="getFocus()" value="Get focus" />
   <input type="button" onclick="loseFocus()" value="Lose focus" />
</p>
<p>
   <a id="myLink" name="myLink" href="http://www.microsoft.com">Microsoft</a>
   <br /><br />
   <input type="button" onclick="changeLink()" value="Change Link" />
</p>
</body>
</html>
