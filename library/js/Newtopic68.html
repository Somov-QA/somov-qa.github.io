Встроенные методы


- вывести сообщение


alert('сообщение');        


- определение ссылок на DOM элементы


       getElementsByTagName

       getElementById

       getElementByClassName


- определение элемента по селектору CSS


       querySelector("#main p:first-child");


- определение все элемента по селектору CSS


       querySelectAll("#main p");


- обработчик событий


       addEventListener('load', clickme, false);


Примеры:


       canvas = document.getElementById('canvas');

       document.querySelectorAll("#main p");

       var list = document.querySelectAll("#main p");

       list[0].onclick = showalert;

       window.onload = init;

       window.addEventListener('load', init, false);
