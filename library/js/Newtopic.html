Наследование с помощью prototype


       var Pet = function(name, legs)

       {

               this.name = name;

       this.legs = legs;

       };


       Pet.prototype.getDetails = function()

       {

               return this.name + ' ' + this.legs;

       };


       var Cat = function(name)

       {

               Pet.call(this.name, 4);

       };


       Cat.prototype = new Pet();

       Cat.prototype.action = function()

{

       return 'событие';

};


обращение к классу


var petCat = Cat('Felix');

var details = petCat.getDetails();

var action = petCat.action();

petCat.name = 'Sylvester';
