HAVING  — Условия отбора в подзапросе


SELECT table1.elem1, AVG(table2.elem1) FROM table1, table2

WHERE table1.elem2 = table2.elem2

GROUP BY table1.elem1

HAVING AVG(table2.elem1) > (

SELECT AVG(table3.elem2) FROM table3

);
