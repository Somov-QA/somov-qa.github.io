INSERT  — Добавление данных в таблицу (добавляем одну строку записи)

INSERT INTO table1 (elem1, elem2, elem3)

VALUE ('text', 150.00, '2017-01-01');


Добавление сразу нескольких строк одним запросом


INSERT INTO table1 (elem1, elem2, elem3)

VALUE (10, 10, 10), (20, 20, 20), (30, 30, 30);


Многострочная инструкция обеспечивает извлечение строк из одной таблицы и добавление из в другую таблицу


INSERT INTO table1 (table1.elem1, table.1.elem2)

SELECT table2.elem1, table2.elem2 FROM table2

WHERE table2.elem3_date < '2008-01-01';
