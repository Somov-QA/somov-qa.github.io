Выражение CASE

Обеспечивает ограничение возможности принятия решения в выражении. (аналог условия)


SELECT elem1,

       CASE        WHEN elem2 > 1000 THEN 'A'

               WHEN elem2 > 5000 THEN 'B'

               ELSE 'C'

       END AS elem2_case

FROM table1


SELECT table1.elem1, table2.elem1 FROM table1, table2

WHERE table1.elem2 = CASE        WHEN (table2.elem2 IS NOT NULL)

                               THEN table2.elem2

                       END


SELECT elem1, elem2,

CAST elem3        WHEN 11 THEN 'A'

                       WHEN 12 THEN 'B'

       END AS elem3_cast

FROM table1
