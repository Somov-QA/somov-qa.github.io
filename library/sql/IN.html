Множество IN


IN - проверка налиция во множестве


SELECT element1 FROM table

WHERE element2 IN (10, 20, 30)


SELECT element1 FROM table

WHERE element2 IN ('New York', 'London')


SELECT element1 FROM table

WHERE element_date IN ('2008-01-04', '2008-01-11', '2008-01-25')
