Диапазон значений BETWEEN


BETWEEN - проверка на принадлежность диапазону


SELECT element1 FROM table

WHERE element_date BETWEEN '2007-10-01' AND '2007-12-31'


SELECT element1 FROM table

WHERE element2 NOT BETWEEN 2000.00 AND 2500.00


SELECT element1 FROM table

WHERE element2 BETWEEN (element1 * 2) AND (element1 * 4)
