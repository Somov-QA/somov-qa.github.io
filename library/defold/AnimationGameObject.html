описание скрипта


local function plane_animation_done(self, url, property)

       local pos = go.get_position()

       pos.x = 800

       go.set_position(pos)

       go.animate(go.get_id(), "position.x", go.PLAYBACK_ONCE_FORWARD, -50, go.EASING_LINEAR, 5, 0, plane_animation_done)

end


function init(self)

   go.animate(go.get_id(), "position.x", go.PLAYBACK_ONCE_FORWARD, -50, go.EASING_LINEAR, 5, 0, plane_animation_done)

end
