* Выражения классов

*/

let MyClass2 = class MyClass2 {

   constructor(){

       console.log('MyClass2', 'Выражения классов');

   }

}


let MySubClass2 = class extends MyClass2 {  // наследование

   constructor(){

       super();

       console.log('MySubClass2', 'Выражения классов');

   }

}


let mc2 = new MyClass2();

let msc2 = new MySubClass2();


console.dir(mc2);

console.dir(msc2);
