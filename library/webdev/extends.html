Наследование классов


/**

* Наследование классов

*/

class Task {

   constructor(title) {

       this._title = title;

       this.done = false;

       Task.staticValue += 1;

       console.log("Основная задача");

   }


   get title(){

       return this._title;

   }


   set title(value){

       this._title = value;

   }


   static getDefaultTitel(){

       return "Задача";

   }


   complete() {

       this.done = true;

       console.log(`Задача "${this._title}" выполнена`);

   }

}


Task.staticValue = 0;


class SubTask extends Task {

   constructor(title, parent) {

       super(title);

       this.parent = parent;

       console.log("Подзадача");

   }


   complete() {

       super.complete();

       console.log(`Подзадача "${this._title}" выполнена`);

   }

}



let task = new Task("Изучить JavaScript");

let subtask = new SubTask("Изучить ES6", task);


task.complete();

subtask.complete();


console.log(task);

console.log(subtask);

console.log(subtask instanceof SubTask);

console.log(subtask instanceof Task);


console.log(SubTask.getDefaultTitel());

console.log(SubTask.staticValue);

console.log(subtask.title);
