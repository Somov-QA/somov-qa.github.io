Классы в ES5


/** ======================================

* Класс

========================================= */

function MyClass(title) {   // конструктор

   this._title = title;

   this._done+ false;

   MyClass.count += 1;

}


Object.defineProperty(MyClass, 'title', {   // геттер и сеттер

   get: function() {

       return this._title;

   },

   set: function(value) {

       this._title = value;

   }

});


MyClass.prototype.complete = function() {   // метод класса

   this._done = true;

}


MyClass.getDefaultTitle = function() {      // статический метод

   return 'Test class';

}


MyClass.count = 0;                          // статическое свойство


/** ======================================

* Подкласс. Наследование.

========================================= */

function MySubClass(title, parent) {

   MyClass.call(this, title);

   this._parent = parent;

}


MySubClass.prototype = Object.create(MyClass.prototype);


MySubClass.prototype.constructor = MySubClass;




// Проверяем -------------------------------

var mc = new MyClass('Создается основной класс');

var msc = new MySubClass('Создается подкласс', mc);

console.log(mc);

console.log(msc);
