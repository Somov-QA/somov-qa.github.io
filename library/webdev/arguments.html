// Параметры по уломчанию

function greet(greeting = 'Hello', name = 'friend'){

   console.log(greeting, name);

}

greet('Hi', 'Bill');

greet('Hi');

// Использование аргументов вместо оставшихся параметров в ES5

function sumES5(){ // ES5

   console.log(arguments);

   console.log(arguments instanceof Array);


   var sum = 0;

   Array.prototype.forEach.call(arguments, function(value){

       sum += value;

   });

   console.log(sum);

}

sumES5(5, 7, 9);



// Оставщиеся параметры

function sumES6(...values){ // ES6

   console.log(values);


   let sum = 0;

   values.forEach(function(value){

       sum += value;

   });

   console.log(sum);

}

sumES6(5, 7, 9);


function sumES6_2(...values){ // ES6

   console.log(

       values.reduce(function(prevValue, currentValue){

           return prevValue + currentValue;

       })

   );

}

sumES6_2(5, 7, 9);
