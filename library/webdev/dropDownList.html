Массив данных dropDownList

Создайте класс в котором опишите массив с данными \modules\admin\components\Cardlist.php


<?php

namespace app\modules\admin\components;


class CardList {

   public static $cardlist = [

       0 => ["id" => "7", "code" => "M001", "name" => "[GU] DERVISH"],

       1 => ["id" => "8", "code" => "M002", "name" => "[GU] SHOGUN"],

       2 => ["id" => "9", "code" => "M003", "name" => "[GU] BERSERKER"],

       ...

       49 => ["id" => "74", "code" => "P014", "name" => "[HFA] MEAN MACHINE"],

   ];

}


Обработка массива данных в контроллере и отправка их в представление


<?php


namespace app\modules\admin\controllers;


use Yii;

use app\modules\admin\models\CardPool;

use app\modules\admin\components\CardList;

use yii\helpers\ArrayHelper;


class CardPoolController extends Controller

{

   public function actionCreate()

   {

       $cardlist = ArrayHelper::map(CardList::$cardlist, 'id', 'name');

       

       $model = new CardPool();


       if ($model->load(Yii::$app->request->post()) && $model->save()) {

           return $this->render('create', [

               'model' => $model,

               'cardlist' => $cardlist,

           ]);

       }

   }

}


Отображение данных в представлении на форме


<?php


use yii\helpers\Html;

use yii\widgets\ActiveForm;

?>


<div class="card-pool-form">


   <?php $form = ActiveForm::begin(); ?>


   <?= $form->field($model, 'profile_id')->textInput(['maxlength' => true]) ?>


   <?= $form->field($model, 'card_id')->textInput()->dropDownList($cardlist) ?>


   <?= $form->field($model, 'count')->textInput() ?>


   <div class="form-group">

       <?= Html::submitButton($model->isNewRecord ? 'Create' : 'Update', ['class' => $model->isNewRecord ? 'btn btn-success' : 'btn btn-primary']) ?>

   </div>


   <?php ActiveForm::end(); ?>


</div>
