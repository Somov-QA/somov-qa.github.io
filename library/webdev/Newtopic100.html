Функциональные компоненты

В папке [scripts] создаем и описываем файл FuncApp.js который по скоей сути App.js

Вместо класса используется чистая функция.


import React from 'react';

import logo from '../logo.svg';

import '../App.css';


const FuncApp = (props) => {

   return (

       <div className="App">

           <header className="App-header">

               <img src={logo} className="App-logo" alt="logo" />

               <h1 className="App-title">Welcome to React</h1>

           </header>

           <p className="App-intro">

               To get started, edit <code>src/App.js</code> and save to reload.

           </p>

       </div>

   );

}


export default FuncApp;


В файле index.js  импортируем новый компонент и в роутере меняем App на  FuncApp



import FuncApp from './scripts/FuncApp';


ReactDOM.render(

   <BrowserRouter>

       <div>

           <Route path="/" component={FuncApp} />

           <Route path="/news" component={News} />

           <Route path="/about" component={About} />

       </div>

   </BrowserRouter>,

   document.getElementById('root')

);

registerServiceWorker();


В результате все работает как прежде, только теперь мы использовали функцию вместо класса.
