COUNT  — Подсчет количества значений в столбце

SELECT COUNT(elem2) FROM table1


Подсчет количества строк в таблице

SELECT COUNT(*) FROM table1


Удаление повторяющихся строк

SELECT COUNT(DISTINCT elem2) FROM table1
