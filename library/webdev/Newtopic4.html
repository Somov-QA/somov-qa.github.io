Установка шаблона в контроллере

Переменная указывает на то какой шаблон должен быть выбран


<?php

namespace app\controllers;

use Yii;

use yii\web\Controller;


class TestController extends Controller {

   

   public $layout = 'main';

   

}

Шаблон для отдельной страницы


class TestController extends Controller {

   

   public $layout = 'main';

   

   public function actionIndex() {

       $this->layout = 'main';

       return $this->render('index', ['message' => 'This is a test!']);

   }

   

}
