Деструктивное присваивание массивов


/**

* Деструктивное присваивание массивов

*/

let lang = ['JavaScript', 'PHP', 'Python', 'Ruby'];

let [js, php, py, rb] = lang;


console.log('Деструктивное присваивание:', js, php, py, rb);


let score = [3, 4, 5, 6];

let [low, mid, high, , empty, test = 13] = score;


console.log('Деструктивное присваивание:', low, mid, high, empty, test);


let [first, ...rest] = score;


console.log('Деструктивное присваивание:', first, rest);


let points = [3, 4, [5, 6]];

let [x, y, point] = points;


console.log('Деструктивное присваивание:', x, y, point);


let [, , [x1, y1]] = points;


console.log('Деструктивное присваивание:', x1, y1);


function computeScore([low, mid]) {

   console.log('Деструктивное присваивание:', low, mid);

}

computeScore([3, 4]);


function getScores() {

   return [10, 20, 30];

}

let scores = getScores();


console.log('Деструктивное присваивание:', scores);


let [l, m, h] = getScores();


console.log('Деструктивное присваивание:', l, m, h);


let yes = 'yes';

let no = 'no';

[yes, no] = [no, yes];


console.log('Деструктивное присваивание: yes=', yes, "  no=", no);
