function applyForVisa(document) {

   console.log('Обработка заявления...');

   let promise = new Promise(function (resolve, reject) {

       setTimeout(function () {

           Math.random() > 0.5 ? resolve({}) : reject('В визе отказано');

       }, 1000);

   });

   return promise;

}

function bookHotel(visa) {

   console.info('Бронь номера в отеле', visa);

   return new Promise(function (resolve, reject) {

       Math.random() > 0.5 ? resolve(visa) : reject('Нет свободных номеров');

   });

}


function buyTickets(booking) {

   console.info('Покупка билетов', booking);

   Math.random() > 0.5 ? Promise.resolve(booking) : Promise.reject('Нет билетов');

}


applyForVisa({})

   .then(visa => {

       console.info('Виза получена', visa);

       return visa;

   })

   .then(bookHotel)

   .then(buyTickets)

   .catch(error => console.error(error));
