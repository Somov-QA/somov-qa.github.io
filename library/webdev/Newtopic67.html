Имя контроллера

Получить и преобразовать имя контроллера (например: card-pool преобразовать в CardPool)


$controller_id = Yii::$app->controller->id;


$controller_search = mb_convert_case($controller_id, MB_CASE_TITLE, "UTF-8");


$controller_search = str_replace("-", "", $controller_search);
