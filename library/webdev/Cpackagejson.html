Cоздать файл package.json

JavaScript ›› Node.js ›› npm ››

Cоздать файл package.json


npm init -y


Чтобы установить все паки описанные в package.json


npm install



Пример структуры файла package.json


{

"name": "project_name",

"title": "ProjectName",

"author": "Somov",

"version": "1.0.0",

"description": "This is a template",

"contributors": [

{

"name": "Evgeniy Somov",

"email": "somov@i.ua"

}

],

"repository": {

"type": "git",

"url": "https://github.com/Somov-QA/templates-phaser-projects"

},

"private": true,

"devDependencies": {

"phaser-ce": "^2.9.4",

"typescript": "^2.0.x",

"node-static": "^0.7.10"

},

"engines": {

"node": "5.x.x"

}

}
