Авторизация

В качастве шаблона есть модель \models\User.php

Наследуется от \yii\base\Object

Интерфейс \yii\web\IdentityInterface


Модель User.php подключается в \config\web.php


'user' => [

   'identityClass' => 'app\models\User',

   'enableAutoLogin' => true,

],



Авторизация происходит в контроллере \controllers\SiteController.php


use yii\filters\AccessControl;


'access' => [

   'class' => AccessControl::className(),

   'only' => ['logout'],

   'rules' => [

       [

           'actions' => ['logout'],

           'allow' => true,

           'roles' => ['@'],

       ],

   ],

],



Проверка пользователя в моделе \models\LoginForm.php

Используется обращение $this->_user = User::findByUsername($this->username);


Запуск проверки в действии actionLogin


if (!Yii::$app->user->isGuest) {

   return $this->goHome();

}


$model = new LoginForm();

if ($model->load(Yii::$app->request->post()) && $model->login()) {

   return $this->goBack();

}

return $this->render('login', [

   'model' => $model,

]);


Разлогиниться actionLogout


Yii::$app->user->logout();

return $this->goHome();


Проверка авторизации в контроллере. Если пользователь авторизоват ему доступно всё.


public function behaviors()

{

   return [

       'access' => [

           'class' => AccessControl::className(),

           'rules' => [

               [

                   'allow' => true,

                   'roles' => ['@'],

               ],

           ],

       ],

       'verbs' => [

           'class' => VerbFilter::className(),

           'actions' => [

               'delete' => ['POST'],

           ],

       ],

   ];

}
