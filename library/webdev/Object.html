function createCar(property, value) {

   /*

   var car = {};

   car[property] = value;

   return car;

   */

   return {

       [property]: value,

       ['_' + property]: value,

       [property.toUpperCase()]: value,

       ['get' + property]() {

           return this[property];

       }

   }

}


let car = createCar('vin', 1);

console.log(car);


// Get и Set ====================================

let person_ES5 = {

   // ES5

   firstName: firstName,

   lastName: lastName,

   email:email,


   sayHello: function(){

       console.log('Hi my name is', firstName, lastName);

   }

};

Object.defineProperty(person_ES5, 'fullName', {

   get: function(){

       return this.firstName + ' ' + this.lastName;

   },

   set: function(value) {

       this.firstName = value;

   }

});


console.log(person_ES5);

console.log(person_ES5.fullName);


let person_ES6 = {

   //ES6

   firstName,

   lastName,

   email,

   sayHello() {

       console.log('Hi my name is', firstName, lastName);

   },

   get fullName(){

       return this.firstName + ' ' + this.lastName;

   },

   set fullName(value){

       this.firstName = value;

   }

};


console.log(person_ES6);

console.log(person_ES6.fullName);
