let browsers = ['Chrome', 'FireFox', 'Opera', 'Safary', 'Edge'];


for(let key in browsers){

   console.log(browsers[key]);

}


for(let value of browsers){

   console.log(value);

}


browsers.forEach(element => {

   console.log(element);

});


console.log(browsers);
