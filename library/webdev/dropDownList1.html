Загрузить данные в dropDownList

Создайте модель работающую с таблицей базы данных Profile.php


<?php

namespace app\modules\admin\models;

use Yii;

class Profile extends \yii\db\ActiveRecord

{

   public static function tableName()

   {

       return 'profile';

   }

}


Обработка данных в контроллере и отправка их в представление


<?php

namespace app\modules\admin\controllers;


use Yii;

use app\modules\admin\models\CardPool;

use app\modules\admin\models\Profile;

use yii\helpers\ArrayHelper;


class CardPoolController extends Controller

{

   public function actionCreate()

   {

       $query = Profile::find()->asArray()->all();

       $profiles = ArrayHelper::map($query, 'id', 'username');

       

       $model = new CardPool();


       if ($model->load(Yii::$app->request->post()) && $model->save()) {

           return $this->redirect(['view', 'profile_id' => $model->profile_id, 'card_id' => $model->card_id]);

       } else {

           return $this->render('create', [

               'model' => $model,

               'profiles' => $profiles,

           ]);

       }

   }

}


Отображение данных в представлении на форме


<?php

use yii\helpers\Html;

use yii\widgets\ActiveForm;

?>


<div class="card-pool-form">


   <?php $form = ActiveForm::begin(); ?>


   <?= $form->field($model, 'profile_id')->textInput(['maxlength' => true])->dropDownList($profiles) ?>


   <?= $form->field($model, 'card_id')->textInput() ?>


   <?= $form->field($model, 'count')->textInput() ?>


   <div class="form-group">

       <?= Html::submitButton($model->isNewRecord ? 'Create' : 'Update', ['class' => $model->isNewRecord ? 'btn btn-success' : 'btn btn-primary']) ?>

   </div>


   <?php ActiveForm::end(); ?>


</div>
