Метатеги

Включение метатегов в виде \views\site\index.php


$this->registerMetaTag([

   'name' => 'keywords',

   'content' => 'ключевые слова'

]);



Включение матегов из контроллера \controllers\SiteController.php


public function actionIndex() {

   $this->view->registerMetaTag([

       'name' => 'keywords',

       'content' => 'ключевые слова'

   ]);

   return $this->render('index');

}
