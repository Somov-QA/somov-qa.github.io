Установка и настройка

JavaScript ›› Node.js ›› npm ›› Webpack ››

Информация


Официальный сайт: https://webpack.js.org/


Описание установки: https://webpack.js.org/guides/installation/


Официальная пошаговая инструкция: https://webpack.js.org/guides/getting-started/



1. Создайте файл package.json


npm init -y



2. Установите webpack


npm install webpack webpack-cli --save-dev



3. После установки пакетов файл package.json будет следующего вида


{

 "name": "test-webpack",

 "version": "1.0.0",

 "description": "Test webpack",

 "main": "index.js",

 "scripts": {

   "test": "echo \"Error: no test specified\" && exit 1"

 },

 "keywords": [],

 "author": "",

 "license": "ISC",

 "devDependencies": {

   "webpack": "^4.5.0",

   "webpack-cli": "^2.0.14"

 },

 "dependencies": {}

}



4. Создайте в папке [src] файл component.js


function component() {

   var element = document.createElement('div');

   element.innerHTML = "<h1>Hello, webpack!</h1><br><h3>This is component.</h3>";

   return element;

}


export default component;



5. Создайте в папке [src] файл index.js


import component from './component';


document.body.appendChild(component());



6. Создайте в папке [dist] файл index.html


<!doctype html>

<html>

<head>

   <title>WEBPACK: Getting Started</title>

</head>

<body>

   <script src="bundle.js"></script>

</body>

</html>



7. Создайте в корне проекта конфигурационный файл webpack.config.js


const path = require('path');


module.exports = {

   entry: './src/index.js',

   output: {

       filename: 'bundle.js',

       path: path.resolve(__dirname, 'dist')

   }

};



8. Чтобы выполнить сборку введите команды (webpack собирает все скрипты по цепочке import импортируемых файлов в скриптах)


npx webpack --config webpack.config.js


В результате в папке [dist] будет создан файл bundle.js


Чтобы упростить сборку можно добавить в файл package.json команду


{

 "name": "modules",

 "version": "1.0.0",

 "description": "Test webpack",

 "main": "index.js",

 "scripts": {

   "test": "echo \"Error: no test specified\" && exit 1",

   "build": "npx webpack --config webpack.config.js"

 },

 "keywords": [],

 "author": "",

 "license": "ISC",

 "devDependencies": {

   "webpack": "^4.5.0",

   "webpack-cli": "^2.0.14"

 }

}
