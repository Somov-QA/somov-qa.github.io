<?php


namespace app\modules\admin\components;


use Yii;

use yii\base\Widget;

use yii\helpers\Url;

use app\modules\admin\components\models\FormProfileSearch;

use app\modules\admin\models\Profile;

use yii\helpers\ArrayHelper;


class ProfileSearch extends Widget {


   public function run() {

       $profiles = Profile::getProfiles();

       

       $model = new FormProfileSearch();

       

       if ($model->load(Yii::$app->request->post()) && $model->validate()) {

           

           if(Yii::$app->request->post('submitUpdate') === 'Update'){

               $profiles = Profile::updateProfileCache();

           }else{

               $profile_id = Yii::$app->request->post('FormProfileSearch')['profile'];

               $controller_id = Yii::$app->controller->id;

           

               $controller_search = mb_convert_case($controller_id, MB_CASE_TITLE, "UTF-8");

               $controller_search = str_replace("-", "", $controller_search);

           

               Yii::$app->response->redirect(Url::to([$controller_id.'/index', $controller_search.'Search[profile_id]' => $profile_id]));

           }

       }


       return $this->render('search', [

           'model' => $model,

           'profiles' => $profiles,

       ]);

   }


}



Строка кода Yii::$app->request->post('submitUpdate') === 'Update' определяет какая кнопка была нажата.

В данном случае если нажата кнопка "Обновить"


Строка кода $profiles = Profile::updateProfileCache(); выполняет обновление кэша данными профилей из базы данных


Строка кода Yii::$app->response->redirect(Url::to([$controller_id.'/index', $controller_search.'Search[profile_id]' => $profile_id])); перенаправляет пользователя на страницу отправляя данные поиска в таблице


Далее идет вызов вида компонента описанного в файле search.php return $this->render('search', ['model' => $model, 'profiles' => $profiles, ]);
