Валидация формы

Валидация выполняется в модели. Так же можно переименовать поля в модели.

Синтаксис правила валидации: [поле, правило]


<?php

namespace app\models\test;

use yii\base\Model;


class TestForm extends Model {

   

   public $name;

   public $email;

   

   public function attributeLabels() {

       return [

           'name' => 'Имя',

           'email' => 'Почта'

       ];

   }

   

   public function rules() {

       return [

           [['name', 'email'], 'required', 'message' => 'Поле обязательно к заполнению'],

           ['email', 'email'],

           ['name', 'string', 'min' => 5],

           ['name', 'string', 'max' => 50],

           ['name', 'string', 'length' => [5, 50]],

       ];

   }

}
