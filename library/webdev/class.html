class MyClass {

   constructor(title){

       this._title = title;  

       MyClass.staticValue = 0;    // static

   }


   get title(){                  // getter

       return this._title;

   }


   set title(value){             // settor

       if(value !== undefined && value !== null)

           this._title = value;

   }


   message(value){                 // method

       console.log(value);

   }


   static default(){           // static

       console.log(MyClass.staticValue);

   }

}


let mc = new MyClass("Hello World!!!");

mc.message(mc.title);

mc.title = "Good by World!!!"

mc.message(mc.title);


MyClass.staticValue = 100;

MyClass.default();


console.log(typeof MyClass);

console.log(typeof mc);

console.log(mc instanceof MyClass);
