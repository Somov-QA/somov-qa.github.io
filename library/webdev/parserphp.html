<?php

if(!isset($_POST['type'])) exit();


if($_POST['type'] === 'dom_document')

{

   echo '<h3>Это обработка страницы с помощью DOMDocument</h3>';

   

   $dom = new DOMDocument();

   $dom->loadHTMLFile('index.php');

   $dom->preserveWhiteSpace = false;

   $table = $dom->getElementsByTagName('table');

   $rows = $table->item(0)->getElementsByTagName('tr');

   foreach ($rows as $row)

   {

       $col = $row->getElementsByTagName('td');

       echo $col->item(0)->nodeValue.'<br>';

       echo $col->item(1)->nodeValue.'<br>';

       echo '<hr>';

   }

}

elseif ($_POST['type'] === 'file_get_html')

{

   echo '<h3>Это обработка страницы с помощью PHP Simple HTML DOM Parser</h3>';

   

   require_once './simple_html_dom.php';

   $page = file_get_html('index.php');

   $links = $page->find('a');

   foreach ($links as $row)

   {

       echo $row->title."<br>";

       echo $row->href."<br>";

       echo '<hr>';

   }

   

}
