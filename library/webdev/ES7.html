* Класс.

========================================= */

class MyClass {

   constructor(title) {

       this._title = title;

       this._done = false;

       MyClass.count += 1;

   }


   get title(){

       return this._title;

   }


   set title(value){

       this._title = value;

   }


   complete() {

       this._done = true;

   }


   static getDefaultTitel(){

       return "Test class";

   }

}

========================================= */

       super(title);

       this._parent = parent;

   }

}
