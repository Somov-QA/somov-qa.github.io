function greet(name) {

   console.log(`Hello ${name}`);

   console.log(`Hello ${name}`.toUpperCase());

   console.log(typeof `Hello ${name}`);

}


greet('Bill');


function createEmail(to, from, subject, message) {

   console.log(`

       To: ${to}

       From: ${from}

       Subject: ${subject}

       Message: ${message}

   `);

}

createEmail('John@mail.com', 'Paul@mail.com', 'Привет', 'ЧТо ты сейчас делаешь?')


function add(x, y){

   if( (typeof x).toString() === "string"){

       console.log(`${x} + ${y} = ${x+y}`);

       console.log(`${x} + ${y} = ${parseInt(x) + parseInt(y)}`);

   }else{

       console.log(`${x} + ${y} = ${x+y}`);

   }

}

add(15, 20);

add('15', '20');
