<?php


namespace app\modules\admin\models;


use Yii;

use yii\base\Model;

use yii\data\ActiveDataProvider;

use app\modules\admin\models\Profile;


/**

* ProfileSearch represents the model behind the search form about `app\modules\admin\models\Profile`.

*/

class ProfileSearch extends Profile

{

   /**

    * @inheritdoc

    */

   public function rules()

   {

       return [

           [['autoinc_id', 'id', 'faction', 'selected_spaceship_config', 'exp', 'card_packs', 'currency', 'medals', 'craft_resources'], 'integer'],

           [['username', 'last_activity', 'selected_deck'], 'safe'],

           [['was_paired'], 'boolean'],

       ];

   }


   /**

    * @inheritdoc

    */

   public function scenarios()

   {

       // bypass scenarios() implementation in the parent class

       return Model::scenarios();

   }


   /**

    * Creates data provider instance with search query applied

    *

    * @param array $params

    *

    * @return ActiveDataProvider

    */

   public function search($params)

   {

       $query = Profile::find();


       // add conditions that should always apply here


       $dataProvider = new ActiveDataProvider([

           'query' => $query,

       ]);


       $this->load($params);


       if (!$this->validate()) {

           // uncomment the following line if you do not want to return any records when validation fails

           // $query->where('0=1');

           return $dataProvider;

       }


       // grid filtering conditions

       $query->andFilterWhere([

           'autoinc_id' => $this->autoinc_id,

           'id' => $this->id,

           'faction' => $this->faction,

           'last_activity' => $this->last_activity,

           'selected_spaceship_config' => $this->selected_spaceship_config,

           'exp' => $this->exp,

           'card_packs' => $this->card_packs,

           'currency' => $this->currency,

           'medals' => $this->medals,

           'craft_resources' => $this->craft_resources,

           'was_paired' => $this->was_paired,

       ]);


       $query->andFilterWhere(['like', 'username', $this->username])

           ->andFilterWhere(['like', 'selected_deck', $this->selected_deck]);


       return $dataProvider;

   }

}
