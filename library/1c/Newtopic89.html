Пример:  запрос который выбирает документы опираясь на справочник


Запрос = СоздатьОбъект("Запрос");

ТекстЗапроса = " //{{ЗАПРОС(Тест)

|Период с ВыбНачПериода по ВыбКонПериода;


|ТоргПредставитель = Справочник.Контрагенты.ТоргПредставитель;

|КонтрСпр = Справочник.Контрагенты.Наименование;


|ДокВН = Документ.ВозвратнаяНакладная.ТекущийДокумент;

|КонтрагентВН = Документ.ВозвратнаяНакладная.Контрагент;

|ТМЦВН = Документ.ВозвратнаяНакладная.ТМЦ;

|ВидТогр = Документ.ВозвратнаяНакладная.ВидТорговли;

|Кво = Документ.ВозвратнаяНакладная.Кво;

|Коэффициент = Документ.ВозвратнаяНакладная.Коэффициент;

|СуммаСНДС = Документ.ВозвратнаяНакладная.СуммаСНДС;

|ДокРНвВН = Документ.ВозвратнаяНакладная.Заказ;


|Функция КвоСумма = Сумма(Кво*Коэффициент);

|Функция СуммаСНДССумма = Сумма(СуммаСНДС);

|Группировка ТМЦВН Без групп;


|Условие(ТоргПредставитель в ВыбТоргПредст);

|Условие(КонтрагентВН в КонтрСпр);

|Условие(ВидТогр в ВыбВидТорг);

|";//}}ЗАПРОС

// Если ошибка в запросе, то выход из процедуры

Если Запрос.Выполнить(ТекстЗапроса) = 0 Тогда

Возврат;

КонецЕсли;
