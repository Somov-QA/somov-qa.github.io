Массив (Array)

Одномерный массив


Массив = Новый Массив;

Массив.Добавить("Привет");

Сообщить(Массив[Массив.Количество()-1]);


Для Счетчик = 0 По 10 Цикл

Массив.Добавить(Счетчик);

КонецЦикла;


Для Каждого ЗначениеМассива Из Массив Цикл

Сообщить(ЗначениеМассива);

КонецЦикла;        

Массив.Очистить();


Ногомерный массив


Матрица = Новый Массив(3,2);

Матрица[0][0] = 1;

Матрица[0][1] = "Пн";

Матрица[1][0] = 2;

Матрица[1][1] = "Вт";

Матрица[2][0] = 3;

Матрица[2][1] = "Ср";

Для Колонка = 0 По Матрица[Строка].Количество()-1 Цикл

Сообщить(Матрица[Строка][Колонка]);
