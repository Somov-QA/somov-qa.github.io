Доступ к группировке запроса


Запрос = СоздатьОбъект("Запрос");

ТекстЗапроса = " //{{ЗАПРОС(Тест)

|КонтрРН=Документ.РасходнаяНакладная.Контрагент;

|ДокРН=Документ.РасходнаяНакладная.ТекущийДокумент;

|Группировка ДокРН;

|Группировка ДокВН;

|Условие (КонтрРН в Контрагент); //Контрагент - это объект "Реквизит диалога"

|"; //}}ЗАПРОС

Пока Запрос.Группировка(1) = 1 Цикл // получаем доступ к первой группировке

Пока Запрос.Группировка(1) = 2 Цикл // получаем доступ к второй группировке

… и т. д.

КонецЦикла;

КонецЦикла;

Запрос.Выполнить(ТекстЗапроса);


или если группировка является предопределенной.


Запрос = СоздатьОбъект("Запрос");

ТекстЗапроса = " //{{ЗАПРОС(Тест)

|КонтрРН=Документ.РасходнаяНакладная.Контрагент;

|ДокРН=Документ.РасходнаяНакладная.ТекущийДокумент;

|Группировка Документ;

|Условие (КонтрРН в Контрагент); //Контрагент - это объект "Реквизит диалога"

|"; //}}ЗАПРОС

Пока Запрос.Группировка(1) = 1 Цикл // получаем доступ к первой группировке "Документ"

Если Запрос.Документ = Запрос.ДокРН Тогда

ДокументРН = Запрос.ДокРН; //присваиваем переменной документ "Расходная накладная"

КонецЕсли;

КонецЦикла;

Запрос.Выполнить(ТекстЗапроса);
