Создаем колонки в "Таблице значений" имена которых присвоены из идентификаторов реквизитов справочника. (произвольно полученные реквизиты)

а так же отмечены периодические реквизиты в "списке значений".


Процедура СформироватьТаблицу(Таблица)

// создаем колонки в ТЗ из реквизитов выбранного справочника.

Если СправочникДляВыгрузки = "" Тогда

Предупреждение("Вы не выбрали справочник!");

СтатусВозврата(0);

Возврат;

КонецЕсли;

Таблица.Очистить(); //очистка таблицы значений.

// создаем новые колонки в таблице значений

Таблица.НоваяКолонка("Объект");

Таблица.НоваяКолонка("Код");

Таблица.НоваяКолонка("Наименование");

Таблица.НоваяКолонка("Родитель");

// определим количество реквизитов в справочнике и запускаем цикл

Для ном = 1 По Метаданные.Справочник(Строка(СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(),""))).Реквизит() Цикл

СтрРекв = Метаданные.Справочник(Строка(СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(), ""))).Реквизит(ном).Идентификатор;

Периодич = Метаданные.Справочник(Строка(СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(), ""))).Реквизит(ном).Периодический; //0- не периодический , 1- периодический реквизит


Таблица.НоваяКолонка(СтрРекв); //создаются колонки всех реквизитов существующих в выбранном справочнике.


//заполним список значений реквизитов и отмечаем периодические.

Если Период = 1 Тогда // значит этот реквизит периодический

СписокПериодРеквизитов.ДобавитьЗначение(СтрРекв);        

СписокПериодРеквизитов.Пометка(ном, 1); // устанавливаем отметку показываюшую периодичность реквизита

Иначе

СписокПериодРеквизитов.ДобавитьЗначение(СтрРекв);        

КонецЕсли;


//запроним список типами реквизитов.

Тип = Метаданные.Справочник(Строка(СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(),""))).Реквизит(ном).Тип; // получаем тип реквизита (число, строка, и. т.д)

СТипов.ДобавитьЗначение(Тип);

КонецЦикла;


//если справочник подчинен дрегому.

Если Фл1 = 1 Тогда //отмечен флажок, значит подчиненный.

Таблица.НоваяКолонка("Владелец");

СписокПериодРеквизитов.ДобавитьЗначение("Владелец");

СТипов.ДобавитьЗначение("Справочник." + СправочникВладелец.ПолучитьЗначение(СправочникВладелец.ТекущаяСтрока(),""));

КонецЕсли;


//определить (Контроль уникальности) кода справочника.

Если Метаданные.Справочник(Строка(СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(),""))).КонтрольУникальности = 1 Тогда

Сообщить("Справочник  [" + СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(),"") + "]  для реквизита КОД установлен КОНТРОЛЬ УНИКАЛЬНОСТИ.");

Сообщить("Это может вызвать ошибку при загрузки данных! Тогда вам нужно будет отключить контроль уникальность.");

КонецЕсли;
