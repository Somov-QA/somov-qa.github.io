Таблица значений

Представляет собой таблицу значений


&НаСервере

Процедура ТабицаНаСервере()

Табл = Новый ТаблицаЗначений;


// Добавить колонку

Табл.Колонки.Добавить("Месяц", Новый ОписаниеТипов("Строка"));

Табл.Колонки.Добавить("Сумма");


// Добавить строки

СтрокаТабл = Табл.Добавить();

СтрокаТабл.Месяц = "Янв";

СтрокаТабл.Сумма = 10000;


СтрокаТабл = Табл.Добавить();

СтрокаТабл.Месяц = "Фев";

СтрокаТабл.Сумма = 20000;


СтрокаТабл = Табл.Добавить();

СтрокаТабл.Месяц = "Фев";

СтрокаТабл.Сумма = 30000;


// Свернуть (сгруппировать)

Табл.Свернуть("Месяц", "Сумма");


// Обход таблицы значений

Для каждого СтрЗнач Из Табл Цикл

               Сообщить(СтрЗнач.Месяц + " : " + СтрЗнач.Сумма);

КонецЦикла;


// Итог по сумме

Сообщить(Табл.Итог("Сумма"));


// Преобразование объекта в реквизит форма

ЗначениеВДанныеФормы(Табл, ТаблицаЗначенийНаФорме);
