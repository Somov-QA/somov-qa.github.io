Показать в агрегатном объекте "Текст" все не периодические константы.


Перем КолКонстант, ИденКонст, СинонимКонст, КоментКонст;

Перем Текст; // объект агрегатного типа данных.

Перем ЗначКонст; // значение константы.

//********************************************


Процедура ПоказатьТекст() // кнопка

Номер = 0;

Текст = СоздатьОбъект("Текст");

КолКонстант = Метаданные.Константа(); // получаем число количества констант.

Для НЦ = 1 По КолКонстант Цикл

ИденКонст = Метаданные.Константа(НЦ).Идентификатор; // идентификатор константы.

СинонимКонст = Метаданные.Константа(НЦ).Синоним; // синоним константы.

КоментКонст = Метаданные.Константа(НЦ).Комментарий; // комментарий константы.

Если Метаданные.Константа(НЦ).Периодический = 0 Тогда // если константа НЕ периодическая мы её отобразим

Номер = Номер + 1;

Текст.ДобавитьСтроку("Данная константа не переодическая...");

Текст.ДобавитьСтроку("ИДЕНТИФИКАТОР__________________");

Текст.ДобавитьСтроку("                   " + ИденКонст);

Текст.ДобавитьСтроку("_______________________________");

Текст.ДобавитьСтроку("СИНОНИМ________________________");

Текст.ДобавитьСтроку("                   " + СинонимКонст);

Текст.ДобавитьСтроку("_______________________________");

Текст.ДобавитьСтроку("КОМЕНТАРИЙ_____________________");

Текст.ДобавитьСтроку("                   " + КоментКонст);

Текст.ДобавитьСтроку("_______________________________");

Текст.ДобавитьСтроку("ЗНАЧЕНИЕ_______________________");

ЗначКонст = Константа.ПолучитьАтрибут(ИденКонст);

Текст.ДобавитьСтроку("                   " + ЗначКонст);

Текст.ДобавитьСтроку("_______________________________");

Текст.ДобавитьСтроку(" ");

Текст.ДобавитьСтроку(Строка(Номер) +") " + "ИДЕНТИФИКАТОР = " + ИденКонст + "СИНОНИМ = " + СинонимКонст + "КОМЕНТАРИЙ = " + КоментКонст + "ЗНАЧЕНИЕ = " + ЗначКонст);

КонецЕсли;

КонецЦикла;

Текст.ТолькоПросмотр();

Текст.Показать("Просмотр констант");

КонецПроцедуры
