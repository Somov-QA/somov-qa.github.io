Использование запроса для отбора данных из документов с последующей передачей их в "Таблицу"


Перем Запрос, ТекстЗапроса, таб;

Если Контр.Выбран()=0 Тогда

Сообщить("Не выбран контрагент");

Возврат;

КонецЕсли;


Запрос=СоздатьОбъект("Запрос");

ТекстЗапроса = " //{{ЗАПРОС(Тест)

|Период с НачДата по КонДата;

|КонтрРН=Документ.РасходнаяНакладная.Контрагент;

|КонтрВН=Документ.ВозвратнаяНакладная.Контрагент;

|КонтрБВ=Документ.БанковскаяВыписка.Субконто;


|ДокРН=Документ.РасходнаяНакладная.ТекущийДокумент;

|ДокВН=Документ.ВозвратнаяНакладная.ТекущийДокумент;

|ДокБВ=Документ.БанковскаяВыписка.ТекущийДокумент;


|СуммаСНДСРН=Документ.РасходнаяНакладная.СуммаСНДС;

|СуммаСНДСВН=Документ.ВозвратнаяНакладная.СуммаСНДС;

|СуммаСНДСБВ=Документ.БанковскаяВыписка.СуммаСНДС;


|функция СуммаРН=Сумма(СуммаСНДСРН);

|функция СуммаВН=Сумма(СуммаСНДСВН);

|функция СуммаБВ=Сумма(СуммаСНДСБВ);


|Группировка Документ;

|Условие (контрРН в контр);

|" ;

Если Запрос.Выполнить(ТекстЗапроса)=0 Тогда

Возврат;

КонецЕсли;


таб=СоздатьОбъект("Таблица");

таб.ИсходнаяТаблица("Таблица1");

Таб.ВывестиСекцию("Заголовок");

Состояние("Заполнение выходной таблицы..."); // Вывести сообщение в строку состояния.

Пока Запрос.Группировка(1) = 1 Цикл    // Обращаемся к греппировке Документ, то есть к перечню выбранных запросом документов

// определяем выводимые документы (Расходная накладная) (Банковская выписка) (Возвратная накладная)

Если Запрос.Документ = Запрос.ДокРН Тогда

// Правило!!! Сначала отправляем значение в ячейку с именем, а потом уже выводим секцию!

ДокументРН = Запрос.ДокРН;  // ДокументРН - это имя ячейки с типом "Выражение", которая полечит Расходную накладную.

Таб.ВывестиСекцию("рн"); // горизонтальная секция с именем "рн"  

КонецЕсли;

Если Запрос.Документ = Запрос.ДокБВ Тогда

// Правило!!! Сначала отправляем значение в ячейку с именем, а потом уже выводим секцию!

ДокументБВ = Запрос.ДокБВ; // ДокументБВ - это имя ячейки с типом "Выражение", которая полечит Банковскую выписку.

Таб.ВывестиСекцию("бв"); // горизонтальная секция с именем "бв"  

КонецЕсли;

Если Запрос.Документ = Запрос.ДокВН Тогда

// Правило!!! Сначала отправляем значение в ячейку с именем, а потом уже выводим секцию!

ДокументВН = Запрос.ДокВН; // ДокументВН - это имя ячейки с типом "Выражение", которая полечит Возвратную накладную.

Таб.ВывестиСекцию("вн"); // горизонтальная секция с именем "вн" 

КонецЕсли;

КонецЦикла;            


// Правило!!! Сначала отправляем значение в ячейку с именем, а потом уже выводим секцию!

СуммаД=(запрос.СуммаВН+запрос.СуммаБВ); // СуммаД - это имя ячейки с типом "Выражение", которая получает значение.

СуммаК=запрос.СуммаРН; // СуммаК - это имя ячейки с типом "Выражение", которая получает значение.
