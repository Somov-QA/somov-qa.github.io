// РАССЧИТАТЬ ОКЛАД ПРИ ПРОВЕДЕНИИ ДОКУМЕНТА ================

Процедура РассчитатьОклады(Регистратор, НаборЗаписей) Экспорт


       Запрос = Новый Запрос;

       Запрос.Текст =

               "ВЫБРАТЬ

               |        НачислениеЗарплатыДанныеГрафика.КоличествоЧасовФактическийПериодДействия КАК ФактическиЧасов,

               |        НачислениеЗарплатыДанныеГрафика.КоличествоЧасовПериодДействия КАК ЗапланированноЧасов,

               |        НачислениеЗарплатыДанныеГрафика.Оклад КАК Оклад,

               |        НачислениеЗарплатыДанныеГрафика.НомерСтроки КАК НомерСтроки

               |ИЗ

               |        РегистрРасчета.НачислениеЗарплаты.ДанныеГрафика(

               |                        Регистратор = &Регистратор

               |                                И ВидРасчета = &ВидРасчета) КАК НачислениеЗарплатыДанныеГрафика";

       

       Запрос.УстановитьПараметр("ВидРасчета", ПланыВидовРасчета.ОсновныеНачисления.Оклад);

       Запрос.УстановитьПараметр("Регистратор", Регистратор);

       

       РезультатЗапроса = Запрос.Выполнить();

       Выборка = РезультатЗапроса.Выбрать();

       

       СтруктураПоиска = Новый Структура;

       СтруктураПоиска.Вставить("НомерСтроки", 0);

       

       Для Каждого Запись Из НаборЗаписей Цикл

               СтруктураПоиска.НомерСтроки = Запись.НомерСтроки;

               

               Если Выборка.НайтиСледующий(СтруктураПоиска) Тогда

                       

                       Если (Выборка.ЗапланированноЧасов = 0) Тогда

                               Запись.Сумма = 0;        

                       Иначе

                               Запись.Сумма = Выборка.ФактическиЧасов / Выборка.ЗапланированноЧасов * Выборка.Оклад;        

                       КонецЕсли;

                       Запись.ЧасовФактическиОтработано = Выборка.ФактическиЧасов;

                       Запись.ЧасовЗапланированно = Выборка.ЗапланированноЧасов;

                       

                       Выборка.Сбросить(); // сбрасываем выборку на начало для следующего поиска

               КонецЕсли;

               

               НаборЗаписей.Записать(, Истина);

               

       КонецЦикла;

       

КонецПроцедуры

// ПЕРЕРАСЧЕТ ПРЕМИЙ ПОСЛЕ ПЕРЕПРОВЕДЕНИЯ ДОКУМЕНТА ================

Процедура Перерасчет() Экспорт


       Запрос = Новый Запрос;

       Запрос.Текст =

               "ВЫБРАТЬ

               |        Перерасчет1.ОбъектПерерасчета КАК ОбъектПерерасчета,

               |        Перерасчет1.Сотрудник КАК Сотрудник

               |ИЗ

               |        РегистрРасчета.НачислениеПремий.Перерасчет1 КАК Перерасчет1

               |ГДЕ

               |        Перерасчет1.ВидРасчета = &ВидРасчета

               |ИТОГИ ПО

               |        ОбъектПерерасчета";

       

       Запрос.УстановитьПараметр("ВидРасчета", ПланыВидовРасчета.ДополнительныеНачисления.Премия);

       ВыборкаПоРегистратору = Запрос.Выполнить().Выбрать(ОбходРезультатаЗапроса.ПоГруппировкам);

       

       Пока ВыборкаПоРегистратору.Следующий() Цикл

               // Получаем ссылку на документ нуждающийся в перерасчёте

               Регистратор = ВыборкаПоРегистратору.ОбъектПерерасчета;        

               

               // Формируем набор записей для обновления данных по Окладу

               НаборЗаписей = РегистрыРасчета.НачислениеЗарплаты.СоздатьНаборЗаписей();

               НаборЗаписей.Отбор.Регистратор.Значение = Регистратор;

               НаборЗаписей.Прочитать();

               

               РасчетЗП.РассчитатьОклад(Регистратор, НаборЗаписей);

               

               // Формируем набор записей для обновления данных по Премии

               НаборЗаписей = РегистрыРасчета.НачислениеПремий.СоздатьНаборЗаписей();

               НаборЗаписей.Отбор.Регистратор.Значение = Регистратор;

               НаборЗаписей.Прочитать();

       

               РасчетЗП.РассчитатьПремию(Регистратор, НаборЗаписей);

               

               // Очистить перерасчитанные записи в перерасчете

               НаборЗаписейПерерасчет = РегистрыРасчета.НачислениеПремий.Перерасчеты.Перерасчет1.СоздатьНаборЗаписей();

               НаборЗаписейПерерасчет.Отбор.ОбъектПерерасчета.Значение = Регистратор;

               НаборЗаписейПерерасчет.Записать();

       КонецЦикла;
