Примеры использования бухгалтерского запроса


1) Оборотно - сальдовая ведомость.

БИ = СоздатьОбъект("БухгалтерскиеИтоги");

БИ.ВыполнитьЗапрос(НачДата, КонДата); // Выполнить отбор и расчет итогов по параметрам

БИ.ВыбратьСчета(); // Открывает выборку счетов, для которых были получены итоги.

Пока БИ.ПолучитьСчет()=1 Цикл // Получить из выборки следующий счет.

Сообщить(БИ.Счет);

// СНД - возвращает дебетовое сальдо по счету на начало периода,

// СНК - возвращает кредитовое сальдо по счету на начало периода

Сообщить("Сальдо на начало по дебету: " + БИ.СНД("С") + " по кредиту: " + БИ.СНК("С"));

// СКД - возвращает дебетовое сальдо по счету на конец периода,

// СКК - возвращает кредитовое сальдо по счету на конец периода.

Сообщить("Сальдо на конец  по дебету: " + БИ.СКД("С") + " по кредиту: " + БИ.СКК("С"));

КонецЦикла;


2) Анализ счета.

БИ = СоздатьОбъект("БухгалтерскиеИтоги");

БИ.ВыполнитьЗапрос(НачДата, КонДата, ВыбСчет,,,3); // 3 - это тип отбираемых итогов (остатки и обороты по счету в целом и обороты между счетами)

// или можно БИ.ВыполнитьЗапрос(НачДата, КонДата, "28.1",,,3);

Если БИ.ПолучитьСчет(,ВыбСчет)=1 Тогда  // или можно Если БИ.ПолучитьСчет(,"28.1")=1 Тогда

Сообщить("Анализ счета: " + ВыбСчет);

Сообщить("Сальдо начальное по дебету: " + БИ.СНД("С") + " по кредиту: " + БИ.СНК("С"));


БИ.ВыбратьКорСчета(); // Открывает выборку корреспондирующих счетов, для которых были получены итоги.

Пока БИ.ПолучитьКорСчет()=1 Цикл // Получить из выборки следующий корреспондирующий счет.

ТекКорСчет = БИ.КорСчет; // Корреспондирующий счет

// КорДО - Возвращает дебетовый оборот между корреспонденциями

// КорКО - Возвращает кредитовый оборот между корреспонденциями

Сообщить("Оборот по счетам: " + ТекКорСчет + " по дебету: " + БИ.КорДО("С") + " по кредиту: " + БИ.КорКО("С"));

КонецЦикла;

Сообщить("Сальдо конечное по дебету: " + БИ.СКД("С") + " по кредиту: " + БИ.СКК("С"));

КонецЕсли;


3) Оборотно сальдовая ведомость по счету. (с детализацией по субконто)

БИ = СоздатьОбъект("БухгалтерскиеИтоги");

БИ.ИспользоватьСубконто("Номенклатура");

БИ.ВыполнитьЗапрос(НачДата, КонДата, "41.1");

БИ.ВыбратьСубконто(1);

Пока БИ.ПолучитьСубконто(1) = 1 Цикл

Сообщить("Товар" + БИ.Субконто(1).Наименование);

Сообщить("Остаток на начало равен " + Строка(БИ.СНД("К")) + "ед. на сумму " + Строка(БИ.СНД("С")));

Сообщить("Остаток на конец равен " + Строка(БИ.СКД("К")) + "ед. на сумму " + Строка(БИ.СКД("С")));

КонецЦикла;


4) Обороты между счетами с детализацией по субконто.

БИ = СоздатьОбъект("БухгалтерскиеИтоги");

БИ.ИспользоватьСубконто("Номенклатура");

БИ.ИспользоватьКорСубконто("Номенклатура");

БИ.ВыполнитьЗапрос(НачДата, КонДата, "41.1", "60.1",,3);

БИ.ВыбратьСубконто(1);

Пока БИ.ПолучитьСубконто(1) = 1 Цикл

Товар = БИ.Субконто(1);

БИ.ВыбратьКорСубконто(1);

Пока БИ.ПолучитьКорСубконто(1) = 1 Цикл

Поставщик = БИ.КорСубконто(1);

Сообщить("Поставщик " + Поставщик + " поставил товар " + Товар + " на сумму " + БИ.КорДО("С"));

КонецЦикла;

КонецЦикла;


5) Получение информации с валютного счет.

БИ = СоздатьОбъект("БухгалтерскиеИтоги");

БИ.ВыполнитьЗапрос(НачДата, КонДата, "52");

БИ.ВыбратьВалюты();

Пока БИ.ПолучитьВалюту() = 1 Цикл

Сообщить("На счете имеется остаток " + БИ.СКД("В") + " " + БИ.Валюта);

КонецЦикла;


6) Промежуточные итоги.

БИ = СоздатьОбъект("БухгалтерскиеИтоги");

БИ.ВыполнитьЗапрос('01.01.2005','31.12.2005', "41.1",,,, "Месяц");

БИ.ВыбратьПериод();

Пока БИ.ПолучитьПериод() = 1 Цикл

Сообщить("За период с " + БИ.НачДата + " по " + БИ.КонДата + " поступило товара на сумму " + БИ.ДО("С"));

КонецЦикла;


7) Извлечение проводок и операций.

БИ = СоздатьОбъект("БухгалтерскиеИтоги");

БИ.ВыполнитьЗапрос(НачДата, КонДата, "66.4,66.44", "91.2",,,"Проводка");

БИ.ВыбратьПериод();

Пока БИ.ПолучитьПериод() = 1 Цикл

Оп = БИ.Операция;

Если (Оп.Дебет.Счет = СчетПоКоду("91.2")) и (Оп.Кредит.Счет = СчетПоКоду("66.4")) или (Оп.Кредит.Счет = СчетПоКоду("66.44")) Тогда

Договор = Оп.Кредит.Субконто(3);

Если Договор.ДатаОкончания > Оп.ДатаОперации Тогда

ПечатьКлиент = Договор.Заимодавец;

ПечатьАгент = Договор.Агент;

ПечатьДоговор = Договор.НомерДок;

ПечатьДата = Договор.ДатаДок;

ПечатьСумма = Оп.Сумма;

Если Оп.Кредит.Счет = СчетПоКоду("66.4") Тогда

ПечатьПроцент = Оп.Сумма;

Иначе

ПечатьПроцент = Оп.ВалСумма;

КонецЕсли;

КонецЕсли;

КонецЕсли;

КонецЦикла;
