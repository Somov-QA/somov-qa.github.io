Создание нового счета, в плане счетов


Сч = СоздатьОбъект("Счет");

Сч.ИспользоватьПланСчетов(ПланыСчетов.Учебный); //Устанавливает план счетов, который будет использован объектом.

Сч.Новый(0); //Создает новый счет (0-счет не имеет субсчета, 1-имеет субсчета) по умолчанию 0

Сч.Код = "44.1";

Сч.Наименование = "Прибыль не распределена";

Сч.ВидСубконто(1, ВидыСубконто.Контрагенты); //Устанавливает/возвращает вид субконто счета

Сч.ВидСубконто(2, ВидыСубконто.Партии); //Устанавливает/возвращает вид субконто счета

Сч.Количественный = 1; // определяем: (1-колич.учет, 0-не колич. учет)

Сч.Валютный = 0; // определяем: (1-валютный, 0-не валютный)

Сч.Забалансовый = 0; // определяем: (1-забалансовый, 0-не забалансовый)

Сч.Активный = 1; // определяем: (1 - счет является активным;2 - счет является пассивным;3 - счет является активно-пассивным)

Сч.Записать();
