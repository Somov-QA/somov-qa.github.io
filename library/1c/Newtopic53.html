Процедура которая наступает в момент выбора "Товара" в табличной части, или с помощью подбора


Процедура Пересчет(ФлагПодбора = 0)

Если Товар.Выбран() = 1 Тогда // "Товар" это поле табличной части данного документа

Если (Форма.ТекущаяКолонка() = "Товар") ИЛИ (ФлагПодбора = 1) Тогда // вводим "Товар"

Цена = Товар.ОтпускнаяЦена.Получить(ДатаДок); // "Цена" это поле табличной части данного документа

Количество = ?(Количество = 0, 1, Количество); // "Количество" это поле табличной части данного документа

Сумма = Цена * Количество; // "Сумма" это поле табличной части данного документа

НДС = Окр(Сумма * Товар.СтавкаНДС.Получить(ДатаДок).Ставка / 100,2,1); // "НДС" это поле табличной части данного документа

Всего = Сумма + НДС; // "Товар" это поле табличной части данного документа

ИначеЕсли Форма.ТекущаяКолонка() = "Количество" Тогда // вводим "Количество"

Если Количество > Ф_О Тогда // Ф_О - это поле табличной части документа (фактический остаток)

Сообщить("Заявка по данной позиции превышает остаток на складе!","!");

Количество = Ф_О;

КонецЕсли;

Сумма = Цена * Количество; // "Сумма" это поле табличной части данного документа

НДС = Окр(Сумма * Товар.СтавкаНДС.Получить(ДатаДок).Ставка / 100,2,1); // "НДС" это поле табличной части данного документа

Всего = Сумма + НДС; // "Товар" это поле табличной части данного документа

ИначеЕсли Форма.ТекущаяКолонка() = "Цена" Тогда // вводим "Цена"

Сумма = Цена * Количество; // "Сумма" это поле табличной части данного документа

НДС = Окр(Сумма * Товар.СтавкаНДС.Получить(ДатаДок).Ставка / 100,2,1); // "НДС" это поле табличной части данного документа

Всего = Сумма + НДС; // "Товар" это поле табличной части данного документа

ИначеЕсли Форма.ТекущаяКолонка() = "Сумма" Тогда // вводим "Сумма"

Цена = ?(Количество = 0, Сумма, Сумма / Количество);

НДС = Окр(Сумма * Товар.СтавкаНДС.Получить(ДатаДок).Ставка / 100,2,1); // "НДС" это поле табличной части данного документа

Всего = Сумма + НДС; // "Товар" это поле табличной части данного документа

ИначеЕсли Форма.ТекущаяКолонка() = "НДС" Тогда // вводим "НДС"

Всего = Сумма + НДС; // "Товар" это поле табличной части данного документа

КонецЕсли;

КонецЕсли;
