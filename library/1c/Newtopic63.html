1C Предприятие 7.7 ›› Запросы ››

Для получения результата запроса используем групировку.


Запрос = СоздатьОбъект("Запрос");

ТекстЗапроса = " //{{ЗАПРОС(Тест)

|Контр = Справочник.Контрагенты.ТекущийЭлемент;

|Группировка Контр;

|"; //}}ЗАПРОС

Если Запрос.Выполнить(ТекстЗапроса) = 1 тогда // выполняем запрос

//Выгружаем результат в таблицу значений.

Запрос.Выгрузить(РезультатЗапроса); //выгружаем результат в таблицу значений "РезультатЗапроса"

Иначе

Предупреждение("Произощла ошибка. Запрос не выполнен.");

КонецЕсли;
