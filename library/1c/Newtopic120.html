Получить все Партии и их значения (цены) из Бухгалтерского итога в режиме запроса


БИ = СоздатьОбъект("БухгалтерскиеИтоги");

БИ.ИспользоватьСубконто(ВидыСубконто.Склады, Склад);

БИ.ИспользоватьСубконто(ВидыСубконто.ТМЦ, ТМЦ);

БИ.ИспользоватьСубконто(ВидыСубконто.Партии);

БИ.ВыполнитьЗапрос(,ДатаДок,"28.1.1");

БИ.ВыбратьСубконто(1);

Пока БИ.ПолучитьСубконто(1) = 1 Цикл

БИ.ВыбратьСубконто(2);

Пока БИ.ПолучитьСубконто(2) = 1 Цикл

БИ.ВыбратьСубконто(3);

Пока БИ.ПолучитьСубконто(3) = 1 Цикл

Цена = БИ.Субконто(3).Цена_Уч;

Партии.ДобавитьЗначение(Строка(БИ.Субконто(3)));

ТекущаяПартия = БИ.Субконто(3);  //эта переменная получает весь элемент партии

КонецЦикла;
