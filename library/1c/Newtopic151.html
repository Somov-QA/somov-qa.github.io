Кнопкой "Подбор" мы в ручную выбираем элементы и заносим в таблицу значений (с произвольно полученными реквизитами)


// ПОДБОР ЭЛЕМЕНТОВ *************************

Процедура Подбор()

ОткрытьПодбор("Справочник." + СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(),""), "ДляПодбора",,1);

КонецПроцедуры


Процедура ОбработкаПодбора(Элемент)

Элемент.ИспользоватьДату(ДатаВыгрузки); //используем дату для периодических реквизитов.

Если (ТЗ.НайтиЗначение(Элемент.Код,,"Код") = 1) И (Фл1 = 0) Тогда

Сообщить("Этот элемент (" + Элемент.Наименование + ") уже занесен в таблицу !");

Иначе

СТР = ТЗ.КоличествоСтрок(); //определяем количество строк в таблице значений

//Создаем новую строку с данными элемента

ТЗ.НоваяСтрока();

ТЗ.Объект = "ЭЛЕМЕНТ";

ТЗ.Код = Элемент.Код;

ТЗ.Наименование = Элемент.Наименование;

ТЗ.Родитель = Элемент.Родитель;


Колонок = ТЗ.КоличествоКолонок(); //определим количество колонок

Для ном = 5 по колонок Цикл //заполняем таблицу с пятой колонки

ТЗ.ТекущаяКолонка(ном);

ИД = ТЗ.ПолучитьПараметрыКолонки(ном); //идентификатор реквизита соответствует имени колонки

// Вставляем значение в таблицу значений из "Элемента" получив доступ к его реквизиту по Идентификатору (ИД)

ТЗ.УстановитьЗначение(СТР + 1, ном, Строка(Элемент.получитьАтрибут(ИД)));

КонецЦикла;


//для подчиненных справочников

Если Фл1 = 1 Тогда

ТЗ.Владелец = Элемент.Владелец;

КонецЕсли;

КонецЕсли;
