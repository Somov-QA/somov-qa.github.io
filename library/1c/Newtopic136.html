Выполнение запросов по регистрам


Использование функций в запросе по регистрам:

Для регистра типа "Обороты" используется только функция Сумма(<внутр_перем>);

Для регистра типа "Остатки" используются функции :

НачОст(<внутр_перем>);

Приход(<внутр_перем>);

Расход(<внутр_перем>);

КонОст(<внутр_перем>);



Пример:  использования запроса для оборотного регистра.

Запрос = СоздатьОбъект("Запрос");

ТекстЗапроса = "";

// установка периода запроса.

Если ДатаКон >= ПолучитьДатуТА() Тогда

ТекстЗапроса = ТекстЗапроса + "Период С ДатаНач;";

Сообщить("конечная дата >= точки актуальности");

Иначе

ТекстЗапроса = ТекстЗапроса + "Период С ДатаНач По ДатаКон;";

Сообщить("конечная дата < точки актуальности");

КонецЕсли;


//текст запроса по продажам.

ТекстЗапроса = ТекстЗапроса + "

|Товар = Регистр.ОборотТовара.Товар;

|Контрагент = Регистр.ОборотТовара.Контрагенты;

|Сумма = Регистр.ОборотТовара.Сумма;

|Количество = Регистр.ОборотТовара.Количество;

|Группировка Товар Без Групп;

|Группировка Контрагент Без Групп;

|Функция ОборотСум = Сумма(Сумма);                      

|Функция ОборотКол = Сумма(Количество);

|";

// Если ошибка в запросе, то выход из процедуры

Если Запрос.Выполнить(ТекстЗапроса) = 0 Тогда

Возврат;

КонецЕсли;


//читаем запрос.

Пока Запрос.Группировка(1) = 1 Цикл

Пока Запрос.Группировка(2) = 1 Цикл

Сообщить(Запрос.Товар);

Сообщить(Запрос.Контрагент);

Сообщить("СУММА:" + Строка(Запрос.ОборотСум));

Сообщить("КОЛИЧЕСТВО:" + Строка(Запрос.ОборотКол));

Сообщить("----------------------------------------");

КонецЦикла;

КонецЦикла;




Пример: использование запроса для регистра остатки.

Запрос = СоздатьОбъект("Запрос");

ТекстЗапроса = "";

// установка периода запроса.

Если ДатаКон >= ПолучитьДатуТА() Тогда

ТекстЗапроса = ТекстЗапроса + "Период С ДатаНач;";

Сообщить("конечная дата >= точки актуальности");

Иначе

ТекстЗапроса = ТекстЗапроса + "Период С ДатаНач По ДатаКон;";

Сообщить("конечная дата < точки актуальности");

КонецЕсли;


//текст запроса по продажам.

ТекстЗапроса = ТекстЗапроса + "

|Товар = Регистр.ОстаткиТоваров.Товар;

|Склады = Регистр.ОстаткиТоваров.Склад;

|Сумма = Регистр.ОстаткиТоваров.Сумма;

|Количество = Регистр.ОстаткиТоваров.Количество;

|Условие(Склады в Склад);

|Группировка Товар Без Групп;

|Функция ПриходСум = Приход(Сумма);

|Функция РасходСум = Расход(Сумма);

|Функция ПриходКол = Приход(Количество);

|Функция РасходКол = Расход(Количество);

|Функция НачОстКол = НачОст(Количество);

|Функция КонОстКол = КонОст(Количество);

|";

// Если ошибка в запросе, то выход из процедуры

Если Запрос.Выполнить(ТекстЗапроса) = 0 Тогда

Возврат;

КонецЕсли;


//читаем запрос.

Пока Запрос.Группировка(1) = 1 Цикл

Сообщить(Запрос.Товар);

Сообщить("ОСТАТОК НА НАЧАЛО:" + Строка(Запрос.НачОстКол));

Сообщить("----------------------------------------");

Сообщить("ПРИХОД СУММА:" + Строка(Запрос.ПриходСум));

Сообщить("ПРИХОД КОЛИЧЕСТВО:" + Строка(Запрос.ПриходКол));

Сообщить("----------------------------------------");

Сообщить("РАСХОД СУММА:" + Строка(Запрос.РасходСум));

Сообщить("РАСХОД КОЛИЧЕСТВО:" + Строка(Запрос.РасходКол));

Сообщить("----------------------------------------");

Сообщить("ОСТАТОК НА КОНЕЦ :" + Строка(Запрос.КонОстКол));

Сообщить("   ");

КонецЦикла;
