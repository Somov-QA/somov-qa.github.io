Перенос данных *.txt


Перенос данных из одной базы данных в другую базу данных при помощи текстового файла. (с помещением элементов в папки)

//-------------------------------------------                                        

Процедура Выгрузка()                                        

   Текст = СоздатьОбъект("Текст");                                        

       Список = СоздатьОбъект("СписокЗначений");                                

       Спр = СоздатьОбъект("Справочник.Сотрудники");                                

       спр.ИспользоватьДату(РабочаяДата());                                

       Спр.ВыборГруппы(1);                                

       Спр.ВыбратьЭлементы(1);                                

       Пока Спр.ПолучитьЭлемент() = 1 Цикл                                

               Если Спр.ЭтоГруппа() = 1 Тогда                        

                       Список.ДобавитьЗначение("Я сам группа");                

                       Список.ДобавитьЗначение(Спр.Код);                

                       Список.ДобавитьЗначение(Спр.Наименование);                

                       //...............                

                       Текст.ДобавитьСтроку(Список.ВСтрокуСРазделителями());                

               Иначе                        

                       Список.ДобавитьЗначение("Я элемент");                

                       Список.ДобавитьЗначение(Спр.Наименование);                

                       Список.ДобавитьЗначение(Спр.Родитель);                

                       //...............                

                       //...............                

                       //...............                

                       //...............                

                       //...............                

                       Текст.ДобавитьСтроку(Список.ВСтрокуСРазделителями());                

               КонецЕсли;                        

                                       

               Список.УдалитьВсе();                        

       КонецЦикла;                                

       Текст.Записать("C:\экспорт_Сотрудники.txt");                                

       Предупреждение("Выгрузка прошла успешно!");

Процедура Загрузка()                                        

       Спр = СоздатьОбъект("Справочник.Сотрудники");                                

       Текст = СоздатьОбъект("Текст");                                

       Список = СоздатьОбъект("СписокЗначений");                                

       Текст.Открыть("C:\экспорт_Сотрудники.txt");                                

                                       

       // Переносим группы (папки) -------------------------------------------                                

       Для Ном1 = 1 По Текст.КоличествоСтрок() Цикл                                

               Стр = Текст.ПолучитьСтроку(Ном1);                        

               Список.ИзСтрокиСРазделителями(Стр);                        

               Если Список.ПолучитьЗначение(1) = "Я сам группа" Тогда                        

                       спр.НоваяГруппа();                

                       спр.Наименование = Список.ПолучитьЗначение(3);                

                       спр.Записать();                

                       // текущая папка                

                       СсылкаНаТекущуюГруппу = Спр.ТекущийЭлемент();                

                                       

                       // Переносим элементы и помещаем их в папки ------------------                

                       Для Ном2 = 1 По Текст.КоличествоСтрок() Цикл                

                               Стр = Текст.ПолучитьСтроку(Ном2);        

                               Список.ИзСтрокиСРазделителями(Стр);        

                               Если (Список.ПолучитьЗначение(1) = "Я элемент") и (Строка(СсылкаНаТекущуюГруппу) = Строка(Список.ПолучитьЗначение(3)))  Тогда        

                                       Спр.Новый();

                                       Спр.ИспользоватьДату(РабочаяДата());

                                       Спр.Наименование = Список.ПолучитьЗначение(2);

                                       Спр.Родитель = СсылкаНаТекущуюГруппу;

                                       //........

                                       //........

                                       //........

                                       //........

                                       //........

                                       Спр.Записать();

                                       // просмотр хода перенисения данных

                                       Сообщить(Строка(СсылкаНаТекущуюГруппу) + "  " + Спр.ТекущийЭлемент() );

                                       

                               КонецЕсли;        

                       КонецЦикла;                

               КонецЕсли;                        

       КонецЦикла;                                

                                       

       Предупреждение("Загрузка прошла успешно!");
