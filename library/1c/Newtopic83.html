Функции


синтаксис: Функция <ИмяФункции> = <ТипФункции> (<Параметр>) [Когда (<Условие>)];

<ТипФункции> - Сумма, Среднее, Минимум, Максимум, Счётчик, <ВнутренняяПеременнаяТипЧисло>


Сумма. В строку итогов будет выведена общая сумма по всем выбранным строкам



// внутр. Переменной "Цена" присваивается "ОтпускнаяЦена" из элемента

// Функция возвращает сумму из внутр. Переменной "Цена"

// В функции можно производить вычисления |Функция СуммаЦена = Сумма(Цена / 2);


Запрос = СоздатьОбъект("Запрос");

ТекстЗапроса = " //{{ЗАПРОС(Тест)

|Период С НачДата По КонДата;

|Тов = Справочник.Товары.ТекущийЭлемент;

|Цена = Справочник.Товары.ОтпускнаяЦена;

|Функция СуммаЦена = Сумма(Цена);

|Группировка Тов Без Групп;

|"; //}}ЗАПРОС

Запрос.Выполнить(ТекстЗапроса);


Среднее, Минимум, Максимум. В строку итогов будет выведено соответствующее значение.


|Функция СуммаЦена = Среднее(Цена);

|Функция СуммаЦена = Минимум(Цена);

|Функция СуммаЦена = Максимум(Цена);


Счётчик. В строке итогов выводится число полученных строк.


Запрос = СоздатьОбъект("Запрос");

ТекстЗапроса = " //{{ЗАПРОС(Тест)

|Период С НачДата По КонДата;

|Тов = Справочник.Товары.ТекущийЭлемент;

|Группировка Тов Без Групп;

|Функция КоличествоЭлементов = Счётчик();

|"; //}}ЗАПРОС

Запрос.Выполнить(ТекстЗапроса);
