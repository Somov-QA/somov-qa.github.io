Процедура ОткрытьФайл()

Перем Имя_Файла, Имя_Папки;

Текст = СоздатьОбъект("Текст");

Список = СоздатьОбъект("СписокЗначений");

//Открываем ранее созданный текстовый файл.

Если ФС.ВыбратьФайл(0,Имя_Файла, Имя_Папки, "ЗАГРУЗКА ДАННЫХ", "Файл БД (*.txt)|*.txt", "*.txt") = 1 Тогда

Текст.Открыть(Имя_Папки + Имя_Файла); //открываем файл в агр. Объекте "Текст"


// Заполняем "таблицу значений"(ТЗ) при помощи "списка значений"(Список)

Для Ном1 = 1 По Текст.КоличествоСтрок() Цикл

Стр = Текст.ПолучитьСтроку(Ном1); //получаем строку из таблицы значений.

Список.ИзСтрокиСРазделителями(Стр); //помещаем полученную строку в список значений

//создаем запись  ------------------------

СТР = ТЗ.КоличествоСтрок();

ТЗ.НоваяСтрока();

ТЗ.Объект = Список.ПолучитьЗначение(1);

ТЗ.Код = Список.ПолучитьЗначение(2);

ТЗ.Наименование = Список.ПолучитьЗначение(3);

ТЗ.Родитель = Список.ПолучитьЗначение(4);


Колонок = ТЗ.КоличествоКолонок(); //определим количество колонок

Для ном = 5 по колонок Цикл //заполняем таблицу с пятой колонки

ТЗ.ТекущаяКолонка(ном);

ИД = ТЗ.ПолучитьПараметрыКолонки(ном); //идентификатор реквизита

ТЗ.УстановитьЗначение(СТР + 1, ном, Строка(Список.ПолучитьЗначение(ном)));

КонецЦикла;
