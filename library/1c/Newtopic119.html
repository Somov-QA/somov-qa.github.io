Процедура Сформировать()                        

       ОДт = 0;                

       ОКт = 0 ;                

                       

       Таб = СоздатьОбъект("Таблица");                

       Таб.ВывестиСекцию("Заголовок");                

                       

       БИ = СоздатьОбъект("БухгалтерскиеИтоги");                

       БИ.ВыполнитьЗапрос(НачДата, КонДата, ВыбСчет,,,3); // 3 - это тип отбираемых итогов (остатки и обороты по счету в целом и обороты между счетами)                

       Если БИ.ПолучитьСчет(,ВыбСчет)=1 Тогда  // или можно Если БИ.ПолучитьСчет(,"28.1")=1 Тогда                

               СНСумДт = БИ.СНД("С");        

               СНСумКт = БИ.СНК("С");        

               СКСумДт = БИ.СКД("С");        

               СКСумКт = БИ.СКК("С");        

               Таб.ВывестиСекцию("Сальдо");        

               Таб.ВывестиСекцию("Шапка");        

                       

                       

                       

               БИ.ВыбратьКорСчета(); // Открывает выборку корреспондирующих счетов, для которых были получены итоги.        

               Пока БИ.ПолучитьКорСчет()=1 Цикл // Получить из выборки следующий корреспондирующий счет.        

                       ВыбСчет = БИ.КорСчет;

                               

                       ОбДт = БИ.ДО("С");

                       ОДт = ОДт + БИ.ДО("С");

                       ОбКт = БИ.КО("С");

                       ОКт = ОКт + БИ.КО("С");

                       Таб.ВывестиСекцию("Данные");

               КонецЦикла;        

       КонецЕсли;
