Использование субконто в операции


// Атрибут  <Субконто>. -----------------------------------------------------------------------------------------------------------------

Операция.НоваяПроводка();  //Создать новую проводку

Операция.Дебет.Счет = СчетПоКоду("64.1", ПланыСчетов.Основной); // определяем счет учета по дебету

Операция.Дебет.<Субконто> = "значение субконто"; // где <Субконто> - название cубконто счета дебета, как оно задано в конфигураторе.

Операция.Кредит.Счет = СчетПоКоду("64.4", ПланыСчетов.Основной); // определяем счет учета по кредиту

Операция.Кредит.<Субконто> = "значение субконто"; // где <Субконто> - название cубконто счета кредита, как оно задано в конфигураторе.

Операция.Записать();


// Метод  Субконто. -----------------------------------------------------------------------------------------------------------------

Операция.НоваяПроводка();  //Создать новую проводку

Операция.Дебет.Счет = СчетПоКоду("64.1", ПланыСчетов.Основной); // определяем счет учета по дебету

Операция.Дебет.Субконто(1, "Значение"); // Обращение к субконто дебета проводки (корреспонденции) операции. Субконто(<НомерСубконто> или <ВидСубконто>,<Субконто>)

Операция.Кредит.Счет = СчетПоКоду("64.4", ПланыСчетов.Основной); // определяем счет учета по кредиту

Операция.Кредит.Субконто(3, "Значение"); // Обращение к субконто кредита проводки (корреспонденции) операции. Субконто(<НомерСубконто> или <ВидСубконто>,<Субконто>)

Операция.Записать();
