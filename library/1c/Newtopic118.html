Процедура Сформировать()                

       СНСДт = 0;        

       СНСКт = 0;        

       СКСДт = 0;        

       СКСКт = 0;        

       ОДт = 0;        

       ОКт = 0 ;        

               

       Таб = СоздатьОбъект("Таблица");        

       Таб.ВывестиСекцию("Заголовок");        

       Таб.ВывестиСекцию("Шапка");        

               

       БИ = СоздатьОбъект("БухгалтерскиеИтоги");        

       БИ.ВыполнитьЗапрос(НачДата, КонДата); // Выполнить отбор и расчет итогов по параметрам        

       БИ.ВыбратьСчета(); // Открывает выборку счетов, для которых были получены итоги.        

       Пока БИ.ПолучитьСчет()=1 Цикл // Получить из выборки следующий счет.        

               ВыбСчет = БИ.Счет;

               // СНД - возвращает дебетовое сальдо по счету на начало периода,

               // СНК - возвращает кредитовое сальдо по счету на начало периода

               СНСумДт = БИ.СНД("С");

               СНСДт = СНСДт + БИ.СНД("С");

               СНСумКт = БИ.СНК("С");

               СНСКт = СНСКт + БИ.СНК("С");

               

               ОбДт = БИ.ДО("С");

               ОДт = ОДт + БИ.ДО("С");

               ОбКт = БИ.КО("С");

               ОКт = ОКт + БИ.КО("С");

               

               // СКД - возвращает дебетовое сальдо по счету на конец периода,

               // СКК - возвращает кредитовое сальдо по счету на конец периода.

               СКСумДт = БИ.СКД("С");

               СКСДт = СКСДт + БИ.СКД("С");

               СКСумКт = БИ.СКК("С");

               СКСКт = СКСКт + БИ.СКК("С");

               Таб.ВывестиСекцию("Данные");

       КонецЦикла;        

       Таб.ВывестиСекцию("Итоги");        

       Таб.Опции(0,0,5,2);        

       Таб.ТолькоПросмотр(1);        

       Таб.Показать("Оборотно сальдовая ведомость.");
