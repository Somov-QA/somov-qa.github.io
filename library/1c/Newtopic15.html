ДеревоЗначений

Дерево значений

Представляет собой дерево значений


&НаСервере

Процедура ТестДеревоЗначений()

ДеревоЗн = Новый ДеревоЗначений;


// Добавить колонки

ДеревоЗн.Колонки.Добавить("Год");

ДеревоЗн.Колонки.Добавить("Месяц");

ДеревоЗн.Колонки.Добавить("Факт");


// Добавить строки

СтрокаДеревоЗн = ДеревоЗн.Строки.Добавить();

СтрокаДеревоЗн.Год = "2017";

СтрокаДеревоЗн.Месяц = "Все месяцы";

СтрокаДеревоЗн.Факт = 100;


// Вложенные строки (подченённая строка)

СтрокаДеревоЗнП = СтрокаДеревоЗн.Строки.Добавить();

СтрокаДеревоЗнП.Месяц = "Янв";


СтрокаДеревоЗнП = СтрокаДеревоЗн.Строки.Добавить();

СтрокаДеревоЗнП.Месяц = "Фев";


// Обход дерева значений

ОбработатьДерево(ДеревоЗн.Строки);


// Преобразование объекта в реквизит форма

ЗначениеВДанныеФормы(ДеревоЗн, ДеревоЗначенийНаФорме);

КонецПроцедуры

Сообщить(СтрЗнач.Год + ":" + СтрЗнач.Месяц);

ОбработатьДерево(СтрЗнач.Строки);

КонецЦикла;
