Автоматическое заполнение ,таблицы значений (с произвольно полученными реквизитами), всеми "Группами" и "Элементами"


Из справочника.

Процедура Заполнить()

спр = СоздатьОбъект("Справочник." + Строка(СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(),"")));

спр.ИспользоватьДату(ДатаВыгрузки); //используем дату для периодических реквизитов.

спр.ВыборГруппы(1); //производим выбор групп

спр.ВыбратьЭлементы();

Пока спр.ПолучитьЭлемент() = 1 Цикл

Если спр.ЭтоГруппа() = 1 Тогда

//создаем запись ГРУППЫ ------------------------

СТР = ТЗ.КоличествоСтрок();

ТЗ.НоваяСтрока();

ТЗ.Объект = "ГРУППА";

ТЗ.Код = спр.Код;

ТЗ.Наименование = спр.Наименование;

ТЗ.Родитель = спр.Родитель;


Колонок = ТЗ.КоличествоКолонок(); //определим количество колонок

Для ном = 5 по колонок Цикл //заполняем таблицу с пятой колонки

ТЗ.ТекущаяКолонка(ном);

ИД = ТЗ.ПолучитьПараметрыКолонки(ном); //идентификатор реквизита

ТЗ.УстановитьЗначение(СТР + 1, ном, Строка(спр.получитьАтрибут(ИД)));

КонецЦикла;

Иначе

//создаем запись ЭЛЕМЕНТА ---------------------

СТР = ТЗ.КоличествоСтрок();

ТЗ.НоваяСтрока();

ТЗ.Объект = "ЭЛЕМЕНТ";

ТЗ.Код = спр.Код;

ТЗ.Наименование = спр.Наименование;

ТЗ.Родитель = спр.Родитель;


Колонок = ТЗ.КоличествоКолонок(); //определим количество колонок

Для ном = 5 по колонок Цикл //заполняем таблицу с пятой колонки

ТЗ.ТекущаяКолонка(ном);

ИД = ТЗ.ПолучитьПараметрыКолонки(ном); //идентификатор реквизита

ТЗ.УстановитьЗначение(СТР + 1, ном, Строка(спр.получитьАтрибут(ИД)));

КонецЦикла;


КонецЕсли;

КонецЦикла;

Из подчиненного справочника

Процедура ЗаполнитьПодчиненныйСпр()

// используем справочник ВЛАДЕЛЕЦ

спрВлад = СоздатьОбъект("Справочник." + Строка(СправочникВладелец.ПолучитьЗначение(СправочникВладелец.ТекущаяСтрока(),"")));

// используем ПОДЧИНЕННЫЙ справочник

спрПодч = СоздатьОбъект("Справочник." + Строка(СправочникДляВыгрузки.ПолучитьЗначение(СправочникДляВыгрузки.ТекущаяСтрока(),"")));


спрВлад.ИспользоватьДату(ДатаВыгрузки);

спрВлад.ВыбратьЭлементы();

Пока спрВлад.ПолучитьЭлемент() = 1 Цикл

спрПодч.ИспользоватьВладельца(спрВлад.ТекущийЭлемент());

спрПодч.ИспользоватьДату(ДатаВыгрузки);

спрПодч.ВыбратьЭлементы();

Пока спрПодч.ПолучитьЭлемент() > 0 Цикл

Если спрПодч.ЭтоГруппа() = 1 Тогда

//создаем запись ГРУППЫ ------------------------
