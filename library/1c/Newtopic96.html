Отправить на печать данные из "Таблицы значений" документа, в "Таблицу"


//событие наступает в момент нажатья кнопки "Печать"

Процедура Печать()

Перем ИтогоНачислено, ВсегоПолучено, к;

ИтогоНачислено = 0;

ВсегоПолучено = 0;

Табл = СоздатьОбъект("Таблица"); //создаем Объект агрег. типа. данных "Таблица"

Табл.ИсходнаяТаблица("ВедомостьЗП"); //Выгрузить заготовленную таблицу по имени "ВедомостьЗП" в сознанный объект "Таблица"


//выводим секцию "Заголовок"

Табл.ВывестиСекцию("Заголовок"); //горизонтальная секция "Заголовок"


//выводим секцию "Шапка"

Табл.ВывестиСекцию("Шапка");

Табл.Область(4,3).Текст = Строка(ДатаДок); //вставляем в ячейку(Строка_4, Столб_3) дату документа.


//выводим секцию "ПереченьСотрудников"

Табл.ВывестиСекцию("ПереченьСотрудников");


//заполнить таблицу значениями

ДокНачЗП = СоздатьОбъект("Документ.НачислениеЗП");

Если ДокНачЗП.НайтиПоНомеру(НомерДок, ДатаДок) = 1 Тогда //находим документ по номеру.

КолСтрокТабЧастиДок = ДокНачЗП.КоличествоСтрок(); //определяем количество строк табличной части данного документа "НачислениеЗП"

Для к = 1 По КолСтрокТабЧастиДок Цикл //читаем строки табличной части документа

ДокНачЗП.ПолучитьСтроку(); //получить следующую строку табличной части из документа "НачислениеЗП"

Табл.ПрисоединитьСекцию("ПереченьСотрудников|Строка"); //Присоединить секцию исходной таблицы-шаблона к результирующей таблице.

Табл.Область(6 + к,1).Текст = Строка(к);

Табл.Область(6 + к,2).Текст = ДокНачЗП.ФИО;

Табл.Область(6 + к,3).Текст = ДокНачЗП.КолОтработанныхДней;

Табл.Область(6 + к,4).Текст = ДокНачЗП.Ставка;

ИтогоНачислено = ИтогоНачислено + ДокНачЗП.Ставка;

Табл.Область(6 + к,5).Текст = ДокНачЗП.ОплатаЗаДень;

Табл.Область(6 + к,6).Текст = ДокНачЗП.Итого;

ВсегоПолучено = ВсегоПолучено + ДокНачЗП.Итого;

КонецЦикла;

КонецЕсли;


//выводим секцию "Подвал"

Табл.ВывестиСекцию("Подвал");

Табл.Область(8 + к,4).Текст = Строка(ИтогоНачислено);

Табл.Область(8 + к,6).Текст = Строка(ВсегоПолучено);

Табл.Опции(0,0,6,0); //настрока отображения таблицы.

Табл.Показать("ВедомостьЗП");  //указываем имя таблицы которую хотим показать.
