package game

{

       import flash.display.Sprite;


       public class Level extends Sprite

       {

                 private var _userID:Int = 0;

               

                 public function Level(x:int, y:int):void

               {

                       super()

                       this.x = x;

                       this.y = y;

               }


               public function get UserID():int

               {

                       return _userID;

               }

               

               public function set UserID(value:int):void

               {

                       _userID = value;

               }

       }

}
