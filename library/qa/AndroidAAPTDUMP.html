# Импорт модулей monkeyrunner для этой программы

from com.android.monkeyrunner import MonkeyRunner, MonkeyDevice

# Для подключения к текущему устройству, возвращает объект MonkeyDevice

device = MonkeyRunner.waitForConnection()

# Устанавливает пакет Android. Обратите внимание, что этот метод возвращает

# логическое значение, так что вы можете проверить результат установки

device.installPackage('c:/myproject/bin/MyApplication.apk')

# Записываем в переменную внутреннее имя пакета

package = 'com.example.android.myapplication'

# Записываем в переменную имя одного из видов деятельности в пакете

activity = 'com.example.android.myapplication.MainActivity'

# Задает имя компонента, чтобы начать с ним работать

runComponent = package + '/' + activity

# Запускает компонент

device.startActivity(component=runComponent)

# Нажимает кнопку Menu

device.press('KEYCODE_MENU', MonkeyDevice.DOWN_AND_UP)

# Снимает скриншот

result = device.takeSnapshot()

# Записывает скриншот в файл

result.writeToFile('c:/myproject/shot1.png','png')

__________________________________________________________________

ПРИМИЧАНИЕ! Файл скрипта должен быть записан в кодировке UTF-8 в противном случае скрипт не будет выполнен.


Чтобы узнать package и activity нужно воспользоваться утилитой aapt

aapt dump badging c:\builds\project.apk
