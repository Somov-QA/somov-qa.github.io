PHPUnit (Быстрый старт)

1. Скачайте и установите Composer https://getcomposer.org/download/

2. Официальный сайт PHPUnit по ссылке https://phpunit.de/


3. Перейдите в раздел Getting Started https://phpunit.de/getting-started/phpunit-7.html


4. Выберите PHP 5.6 в разделе https://phpunit.de/getting-started/phpunit-5.html


5. Установите сервер WAMP (или другой) http://www.wampserver.com/ru/


6. Создайте на сервере папку C:\wamp64\www\test-phpunit


7. Откройте окно консоли и введите команду установки PHPUnit


composer require --dev phpunit/phpunit ^5


8. После того как установка будет завершена перейдите в папку C:\wamp64\www\test-phpunit\vendor\bin  и введите команду


phpunit --version

Если установка прошла корректно вы увидите информацию о версии PHPUnit


9. Создайте в папке C:\wamp64\www\test-phpunit\vendor\bin два скрипта:


файл: Calculator.php


<?php


class Calculator

{

   public function add($a, $b)

   {

       return $a + $b;

   }

}



файл: CalculatorTest.php


<?php


require 'Calculator.php';

class CalculatorTests extends PHPUnit_Framework_TestCase

{

   private $calculator;

   protected function setUp()

   {

       $this->calculator = new Calculator();

   }

   protected function tearDown()

   {

       $this->calculator = NULL;

   }

   public function testAdd()

   {

       $result = $this->calculator->add(1, 2);

       $this->assertEquals(3, $result);

   }

}



10. Запуск теста. Для этого перейдите в папку  C:\wamp64\www\test-phpunit\vendor\bin и введите команду


phpunit CalculatorTest.php



Официальный пример с сайта разработчика.


1. Отредактируйте файл Composer.json


{ "autoload": { "classmap": [ "src/" ] }, "require-dev": { "phpunit/phpunit": "^5" } }


2. Создайте папку  C:\wamp64\www\test-phpunit\src


файл: Email.php


<?php

final class Email

{

   private $email;


   private function __construct($email)

   {

       $this->ensureIsValidEmail($email);


       $this->email = $email;

   }


   public static function fromString($email)

   {

       return new self($email);

   }


   public function __toString()

   {

       return $this->email;

   }


   private function ensureIsValidEmail($email)

   {

       if (!filter_var($email, FILTER_VALIDATE_EMAIL)) {

           throw new InvalidArgumentException(

               sprintf(

                   '"%s" is not a valid email address',

                   $email

               )

           );

       }

   }

}


2. Создайте папку  C:\wamp64\www\test-phpunit\tests


файл: EmailTest.php


<?php

use PHPUnit\Framework\TestCase;


final class EmailTest extends TestCase

{

   public function testCanBeCreatedFromValidEmailAddress()

   {

       $this->assertInstanceOf(

           Email::class,

           Email::fromString('user@example.com')

       );

   }


   public function testCannotBeCreatedFromInvalidEmailAddress()

   {

       $this->expectException(InvalidArgumentException::class);


       Email::fromString('invalid');

   }


   public function testCanBeUsedAsString()

   {

       $this->assertEquals(

           'user@example.com',

           Email::fromString('user@example.com')

       );

   }

}


3. Запустите тест. Для этого перейдите в папку  C:\wamp64\www\test-phpunit\vendor\bin и введите команду


phpunit --bootstrap C:\wamp64\www\test-phpunit\vendor\autoload.php C:\wamp64\www\test-phpunit\tests\EmailTest
