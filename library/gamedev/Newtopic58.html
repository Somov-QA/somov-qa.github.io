function loadAtlas()

{

       var loader = new PIXI.loaders.Loader();

       loader.add('imageAtlas',"atlas.json");

       loader.once('complete',onLoadedComplete);

       loader.load();

}


function onLoadedComplete(loader, res)

{

       var imageAtlas = PIXI.Texture.fromFrame('image.png');

}
