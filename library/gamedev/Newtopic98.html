Анимация спрайта


private animatedDice: Phaser.Sprite;

private anim: Phaser.Animation;

private createAnimatedDice(){

      let randomFrame: number = Utils.getRandomRange(1, 16);

      let frameName: string;

      if(randomFrame < 10){

             frameName = 'button/roll0' + randomFrame + '.png';

      }else{

             frameName = 'button/roll' + randomFrame + '.png';

      }

      this.animatedDice = this.game.add.sprite(0, 0, Atlases.Dice, frameName);

      this.addChild(this.animatedDice);

      let animationFrameList:string[] = [];

      for (let i = 0; i < 16; i++) {

             let frameNum:string = (i+1).toString();

             if (i+1 < 10) {

                    frameNum = "0"+ (i+1).toString();

             }

             animationFrameList.push("button/roll" + frameNum + ".png");

      }

      this.anim = this.animatedDice.animations.add("rollButtonAnimation", animationFrameList, 0, true);

      this.events.onInputOver.add(()=>{

             this.playAnimation();

      }, this);


      this.events.onInputOut.add(()=>{

             this.stopAnimation();

      }, this);

}


public playAnimation(){

      if(this.animatedDice){

             this.anim.play();

      }

}


public stopAnimation(){

      if(this.animatedDice){

             this.anim.stop();

      }

}


public removeAnimatedDice(){

      if(this.animatedDice){

             this.animatedDice.kill();

      }

}
