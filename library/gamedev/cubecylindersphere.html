Примитивы (cube, cylinder, sphere)

Игровые движки ›› ThreeJS ››

var cubeGeometry = new THREE.CubeGeometry(100, 100, 100);

var cubeMaterial = new THREE.MeshNormalMaterial();

var cube = new THREE.Mesh(cubeGeometry, cubeMaterial);

cube.position.y = 300;

scene.add(cube);


var cylinderGeometry = new THREE.CylinderGeometry(50, 50, 200);

var cylinderMaterial = new THREE.MeshNormalMaterial();

var cylinder = new THREE.Mesh(cylinderGeometry, cylinderMaterial);

cylinder.position.x = -200;

scene.add(cylinder);


var sphereGeometry = new THREE.SphereGeometry(100);

var sphereMaterial = new THREE.MeshNormalMaterial();

var sphere = new THREE.Mesh(sphereGeometry, sphereMaterial);

sphere.position.x = 200;

scene.add(sphere);
