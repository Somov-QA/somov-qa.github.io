Унарная длина #


s = "donkey"

print(#s)                                -- 6

t = { "a", "b", "c", "d" }

print(#t)                                -- 4

u = { a = 1, b = 2, c = 3 }

print(#u)                                -- 0

v = { "a", "b", nil }

print(#v)                                -- 2
