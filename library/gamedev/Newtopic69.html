var windowStage;


function windowMaskCreate(widthWindow, heighWindow)

{

       windowStage = new PIXI.Container();

       

       var posX = (widthWindow - 800) / 2;

       var posY = (heighWindow - 600) / 2.5;

       

       var thing = new PIXI.Graphics();

       thing.position.x = 0;

       thing.position.y = 0;

       thing.lineStyle(0);

       thing.clear();

       thing.beginFill(0x8bc5ff, 0.4);

       thing.moveTo(posX, posY);

       thing.lineTo(posX, posY);

       thing.lineTo(posX + 800, posY);

       thing.lineTo(posX + 800, posY + 600);

       thing.lineTo(posX, posY + 600);

       

       windowStage.mask = thing;

       

       stage.addChild(windowStage);

}


function windowRemove()

{

       stage.removeChild(windowStage);

       windowStage = null;

}
