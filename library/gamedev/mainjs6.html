var game;


function onInit()

{

       game = new Phaser.Game(860, 730, Phaser.AUTO, "", { create: create });

       game.state.add("menuGame", menuGame);

}


window.addEventListener("load", onInit, false);


function create()

{

       game.load.onLoadStart.add(loadStart, this);

           game.load.onFileComplete.add(fileComplete, this);

           game.load.onLoadComplete.add(loadComplete, this);


       game.load.image('background', './assets/background.jpg');

       game.load.image('logo', './assets/logo.png');

       game.load.image('button', './assets/button.png');

       game.load.atlas('button_atlas', 'assets/button_atlas.png', 'assets/button_atlas.json');


       game.load.start();

}


function loadStart()

{


}


function fileComplete(progress, cacheKey, success, totalLoaded, totalFiles)

{

       console.log("File Complete: " + progress + "% - " + totalLoaded + " out of " + totalFiles);

}


function loadComplete()

{

       game.state.start("menuGame");

}
