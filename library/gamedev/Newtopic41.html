var stage = new PIXI.Container();


function loadTexture()

{

var loader = new PIXI.loaders.Loader();

loader.add('bdImage',"./assets/image/background.jpg");

loader.once('complete',onLoaded);

loader.load();

}


function onLoaded(loader, res)

{

var windowSprite = new PIXI.Sprite(res.bdImage.texture);

windowSprite.position.x = 0;

windowSprite.position.y = 0;

windowSprite.scale.set(0.9);


stage.addChild(windowSprite);

}
