Изменение анимации в MovieClip

Отключение интерактивности

var atlas = new Object();

var anim;


function initAnimation()

{

       anim = new PIXI.extras.MovieClip(atlas["anim1"]);

       anim.loop = false;

       anim.animationSpeed = 0.2;

       anim.onComplite = onAnimationComplete;

       anim.play

       stage.addChild(anim);

}


function onAnimationComplete()

{

       anim.stop();

       updateAnimation();

}


function updateAnimation()

{

       anim.texture = atlas["anim2"];

       anim.gotoAndPlay(0);

}
