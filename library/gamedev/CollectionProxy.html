function on_message(self, message_id, message, sender)

   if message_id == hash("proxy_loaded")

and sender.fragment == hash("menu_collection_proxy") then  

-- если прокси коллекция загружена

       -- инициализируем и включаем загруженную прокси коллекцию.

       msg.post(sender, "init")        -- инициализация загруженной прокси коллекции

       msg.post(sender, "enable") -- включение загруженной прокси коллекции

       msg.post(sender, "acquire_input_focus") -- передаём фокус в загруженную коллекцию

end
