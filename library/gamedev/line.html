var lineGeometry = new THREE.Geometry();                                // геометрия

lineGeometry.vertices.push(new THREE.Vector3(0,0,0));        // вершина (начальное положение) (вектор)

lineGeometry.vertices.push(new THREE.Vector3(0,200,0));        // вершина

lineGeometry.vertices.push(new THREE.Vector3(30,200,0));        // вершина

                               

var lineMaterial = new THREE.LineBasicMaterial({color:0xFF0000, linewidth:1, opacity:0.5}); // материал

                               

var line = new THREE.Line(lineGeometry, lineMaterial); // линия

                               

scene.add(line); // добавляем на сцену
