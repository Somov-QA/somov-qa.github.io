Для корректной работы необходимо использовать CANVAS.


module GameName{

      export class Game extends Phaser.Game {

      constructor() {

             super({

                    enableDebug: true, // включить дебаг

                    width: Constants.GAME_WIDTH,

                    height: Constants.GAME_HEIGHT,

                    renderer: Phaser.CANVAS, // необходима канва для использования дебага

                    parent: 'content',

                    transparent: true,

                    antialias: true

             });

      }

}


Использование отладки во время работы с физикой


private sprite: Phaser.Sprite;

public create(): void {

      this.game.physics.startSystem(Phaser.Physics.ARCADE);

      this.game.physics.arcade.enable(this.sprite);

}

public render(): void {

      this.game.debug.body(this.sprite);

      this.game.debug.bodyInfo(this.sprite, 32, 32);

}
