Пример: Использование групп для построения колоды и стека в карточной игре Солитёр

иаграмма взаимодействия групп между собой и с другими объектами


1. Объявление переменных: общая группа для колоды и стека, группа для колоды и колода.

private stackAndDeckGroup:Phaser.Group;

private deckGroup:Phaser.Group;

private deck:CardDeck;


2. Создание группы и их объекты

public create(): void {

      //create deck and stack group

      var cardsColNumInGroup:number = 7;

      this.stackAndDeckGroup = this.game.add.group();

      for (var i:number = 0; i < cardsColNumInGroup; i++) {

             let eachColGroup:Phaser.Group = this.game.add.group();

             this.stackAndDeckGroup.add(eachColGroup);

             let cardColBG:Phaser.Sprite = this.game.add.sprite(0, 0, Atlases.AllArts, "back_square_02.png", eachColGroup);

             this.setFrameAndAlpha(i, cardColBG);

             eachColGroup.x = (cardColBG.width + this.cardsSpaceBetweenCol) * i;

      }

      //create deck group

      this.deckGroup = this.game.add.group();

      var deckBG:Phaser.Sprite = this.game.add.sprite(0, 0, Atlases.AllArts, "back_square_01.png", this.deckGroup);

      this.deck = new CardDeck(this.game, 0, 0, Atlases.AllArts, "card_deck_00.png", this.deckGroup, this.cardFlipFrameList);

      this.deck.setDeckFrame(24);

      this.deck.inputEnabled = true;

      this.deck.events.onInputUp.add(this.onClickDeck, this);

}

4. Описание объекта CardDeck

module Fabrique{

      export class CardDeck extends Phaser.Sprite {

             private spriteGroup:Phaser.Group;

             constructor(game:Phaser.Game, x:number, y:number, key:string, frame:string, group:any, cardFlipFrameList:string[]){

                    super(game, x, y, key, frame);

                    this.game = game;

                    this.spriteGroup = group;

                    this.init();

             }

             private init():void{

                    this.spriteGroup.add(this);

                    var bgSprite:any = this.spriteGroup.getAt(0);

             }

      }

}
