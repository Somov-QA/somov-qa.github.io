Класс: комбинированное наследование

Языки программирования ›› JavaScript ››

function Class1(name)

{

       this.name = name;

       this.color = ["red", "blue"];

}


Class1.prototype.getName = function()

{

       return this.name

};


function Class2(name, age)

{

       Class1.call(this, name);        // наследование свойств

       this.age = age

}


Class2.prototype = new Class1();        // наследование методов

Class2.prototype.getAge = function()

{

       return this.age

};


ИСПОЛЬЗОВАНИЕ


var cl1 = new Class2("John", 29);

cl1.color.push("black");                // red, blue, black

cl1.getName();                                // John

cl1.getAge();                                // 29


var cl2 = new Class2("Greg", 32);

alert(cl2.color);                                // red, blue

cl2.getName();                                // Greg

cl2.getAge();                                // 32

cl2 = null;                                        // undefined
