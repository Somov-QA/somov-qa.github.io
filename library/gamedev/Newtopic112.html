var starsGeometry = new THREE.Geometry();

var starMaterial = new THREE.PointsMaterial( { color: 0x888888 } );


for(var i = 0; i < 10000; i++)

{

var vertex = new THREE.Vector3();

vertex.x = THREE.Math.randFloatSpread(10000);

vertex.y = THREE.Math.randFloatSpread(10000);

vertex.z = THREE.Math.randFloatSpread(10000);

starsGeometry.vertices.push(vertex);

}


var starsParticles =  new THREE.Points( starsGeometry, starMaterial);

scene.add(starsParticles);
