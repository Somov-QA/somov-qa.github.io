var windowanimMovieClip;


function windowanimMovieClipCreate(posX, posY, textureArray)

{

       

       windowanimMovieClip = new PIXI.extras.MovieClip(textureArray);

       windowanimMovieClip.position.x = posX;

       windowanimMovieClip.position.y = posY;

       windowanimMovieClip.loop = false;

       windowanimMovieClip.animationSpeed = 0.2;

       windowanimMovieClip.onComplete = onwindowanimAnimationComplete;

       windowanimMovieClip.play();

       

       windowStage.addChild(windowanimMovieClip);

}


function onwindowanimAnimationComplete()

{

       windowanimMovieClip.stop();        // или windowStage.removeChild(this);

}


function AnimationUpdate(textureArray)

{

       windowanimMovieClip.textures = textureArray;

       windowanimMovieClip.gotoAndPlay(0);

}
