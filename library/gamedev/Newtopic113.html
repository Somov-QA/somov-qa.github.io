Переменные и типы

Функции

nil - отсутствие заначения

print(my_var)


boolean - логический тип

flag = true



number - число в виде 64-разрядных целых чисел или 64 чисел с плавающей точкой разрядных.

print(10) --> prints '10'

print(10.0) --> '10'

print(10.000000000001) --> '10.000000000001'


string - строка

my_string = "hello"

another_string = 'world'

print("one\nstring")


локальные переменные

local x = 1
