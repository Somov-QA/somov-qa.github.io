var renderer;

var stage;


function init()

{

       renderer = PIXI.autoDetectRenderer(860, 730,{backgroundColor : 0xFFFFFF});

       document.body.appendChild(renderer.view);

       stage = new PIXI.Container();


       draw();


       // тут вызывается функция начала всей игры

       // startGame();

}


window.addEventListener("load", init, false);


function draw()

{

       requestAnimationFrame(draw);

       renderer.render(stage);

}
