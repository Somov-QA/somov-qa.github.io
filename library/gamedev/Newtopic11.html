Четность / Нечетность

Игровые движки

Определить чётное или не чётное значение

private getEvenOrOdd(n:number):boolean {

      if(n & 1) {

             return false; // odd (нечетное число)

      } else {

             return true; // even (четное число)

      }

}

private rowOffset:number = 0;
