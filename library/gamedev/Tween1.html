Анимация - Tween

Примеры кода

var dice: Phaser.Sprite = new Phaser.Sprite(this.game, 305, 200, 'dice_win_atlas', 'dicewin00');

dice.animations.add('dice', Phaser.Animation.generateFrameNames('dicewin', 0, 3, '', 2), 15, true);

dice.animations.play('dice');

this.stage.addChild(dice);


var tween: Phaser.Tween;

tween = this.game.add.tween(dice);

tween.to({ y: 560 }, 2000, 'Linear');

tween.onComplete.add(this.onMoveDownComplete, this);

tween.start();


onMoveDownComplete(event){

       event.animations.stop('dice');

event.animations.frameName = 'dicewin02';

}


<static> generateFrameNames(prefix, start, stop, suffix, zeroPad) → {Array.<string>}

prefix - префикс кадра 'dicewin'

start - индекс начального кадра анимации

stop - индекс последнего кадра анимации

suffix - суффикс если есть, например подчёркивание _

zeroPad - индекс кадра отображаемого при отключенной анимации
