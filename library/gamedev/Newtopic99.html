Блокирующее окно

var group: Phaser.Group = new Phaser.Group(this.game, this.stage);



var sprite: Phaser.Sprite = new Phaser.Sprite(this.game, 50, 50, 'image');

this.group.addChild(sprite);


var button1: Phaser.Button = new Phaser.Button(this.game, 50, 150, 'button_image', this.onButtonClick, this);

button1.name = 'start_button';

button1.onInputOver.add(this.onButtonOver, this);

button1.onInputOut.add(this.onButtonOut, this);

this.group.addChild(button1);


/*


Добавляем разделяющее поле которое блокирует нажатие кнопок находящихся под ним

      (кнопка button1 становится не активной)

*/


var graphicOverlay: Phaser.Graphics = new Phaser.Graphics(this.game, 0, 0);

graphicOverlay.beginFill(0x000000, 0.5);

graphicOverlay.drawRect(0, 0, this.game.width, this.game.height);

graphicOverlay.endFill();


var image: Phaser.Image = new Phaser.Image(this.game, 0, 0, graphicOverlay.generateTexture());

image.inputEnabled = true;

this.group.addChild(image);


/* Добавляем кнопку которая будет активна */


var button2: Phaser.Button = new Phaser.Button(this.game, 250, 50, 'button_back', this.onButtonClick, this);

button2.name = 'button_back';

button2.onInputOver.add(this.onButtonOver, this);

button2.onInputOut.add(this.onButtonOut, this);

this.group.addChild(button2);



this.stage.addChild(this.group);
