var game;

var sprite;

var bounds;


function onInit()

{

       game = new Phaser.Game(800, 600, Phaser.CANVAS, 'phaser-example', {preload: preload, create: create, update: update, render: render });

}


window.addEventListener("load", onInit, false);


function preload()

{

   game.load.image('boxTexture', './assets/texture.jpg');

}


function create()

{

   bounds = new Phaser.Rectangle(100, 100, 500, 400);


   var graphics = game.add.graphics(bounds.x, bounds.y);

   graphics.beginFill(0x000077);

   graphics.drawRect(0, 0, bounds.width, bounds.height);


   sprite = game.add.sprite(300, 300, 'boxTexture');

   sprite.scale.setTo(0.2, 0.2);

   sprite.inputEnabled = true;

   sprite.anchor.set(0.5);


   sprite.input.enableDrag();                // включаем способность перетаскивания

   sprite.input.boundsRect = bounds;        // перетаскивание в пределах данной области

}
