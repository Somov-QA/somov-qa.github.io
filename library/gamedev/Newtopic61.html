function loadAtlasAnimation()

{

       var loader = new PIXI.loaders.Loader();

       loader.add('animAtlas',"test_animation.json");

       loader.once('complete',onLoadedComplete);

       loader.load();

}


function onLoadedComplete(loader, res)

{

       var dv_AnimTexturesArray = [];

       dv_AnimTexturesArray = loadAnimationTextures(11, 'dv_');

}


function loadAnimationTextures(countFrame, nameFrame)

{

       var nameTexture;

       var animTextures = [];

       for(var i = 1; i <= countFrame; i++)

       {

               if(i < 10)

               {

                       nameTexture = nameFrame + '0' + i + '.png'

               }else{

                       nameTexture = nameFrame + i + '.png'

               }

               var texture = PIXI.Texture.fromFrame(nameTexture);

               animTextures.push(texture);

       }

       return animTextures;

}
