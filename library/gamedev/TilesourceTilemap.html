Сообщение “set_tile” - позволяет менять текстуры в клитках


msg.post(url, “set_tile”, { layer_id, position, tile, dx, dy})


       layer_id - идентификатор слоя

       position - положение ячейки которую нужно изменить

       tile - индекс плитки (0 - очистить, по умолчанию, 1 - для первой плитки)

       dx - горизонтальное смещение (по умолчанию 0)

       dy - вертикальное смещение  (по умолчанию 0)

function init(self)

       msg.post(".", "acquire_input_focus")

end


function on_input(self, action_id, action)

       if action_id == hash("touch") then

               local pos = vmath.vector3(action.x, action.y - 64, 0)

               msg.post("/test_tile#tilemap", "set_tile", {layer_id = hash("layer1"), tile = 2, position = pos, dy = 1})

       end

end


Примеры сообщений для работы с тайлами


Координаты указываются через GameObject

function on_message(self, message_id, message, sender)

       if message_id == hash("edit_tile") then

               msg.post("/test_tile#tilemap", "set_tile", {layer_id = hash("layer1"), position = go.get_position(), dy = 1})

-- или

               msg.post("/test_tile#tilemap", "set_tile", {layer_id = hash("layer1"), tile = 1, position = go.get_position(), dy = 1})

       end

end


Указываем координаты в ручную

function on_message(self, message_id, message, sender)

       if message_id == hash("edit_tile") then

               local doorpos = vmath.vector3(430, 365, 0)

               msg.post("/test_tile#tilemap", "set_tile", {layer_id = hash("layer1"), tile = 0, position = doorpos, dy = 1})

               doorpos = vmath.vector3(500, 360, 0)

               msg.post("/test_tile#tilemap", "set_tile", {layer_id = hash("layer1"), tile = 0, position = doorpos, dy = 1})

           end

end
