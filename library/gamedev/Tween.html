var game;

var dice;


function onInit()

{

       game = new Phaser.Game(860, 730, Phaser.CANVAS, '', { preload: preload, create: create, update: update, render: render });

}


window.addEventListener("load", onInit, false);



function preload()

{

   game.load.atlas('dice_atlas', 'assets/dice_atlas.png', 'assets/dice_atlas.json');

}



function create()

{

   dice = game.add.sprite(0, 0, 'dice_atlas', 'dice09');

   dice.animations.add('dice', Phaser.Animation.generateFrameNames('dice', 0, 10, '', 2), 15, true);
   dice.animations.play('dice');


   startTween();

}


function startTween()

{

   var tween = game.add.tween(dice);


   tween.to({ y: game.world.height - dice.height }, 1000 + Math.random() * 3000, Phaser.Easing.Bounce.In);

   tween.to({ y: 20 }, 1000, Phaser.Easing.Bounce.In);

   tween.onComplete.add(startTween, this);

   tween.start();

}
