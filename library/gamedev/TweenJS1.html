TweenJS (Перемещение объекта)

Загрузка и воспроизведение звуков

var stage = new PIXI.Container();

var windowSprite;

var windowStage;


function createWindow()

{

windowSprite = new PIXI.Sprite(bgTexture);

windowSprite.name = "windowSprite";

windowSprite.position.x = 800;

windowSprite.position.y = (600 / 3.6);

stage.addChild(windowSprite);


windowStage = new PIXI.Container();

windowStage.addChild(windowSprite);

stage.addChild(windowStage);


startTween();

}


function startTween()

{

createjs.Tween.get(windowSprite, {loop: false})

.to({x: 580}, 1000, createjs.Ease.getPowInOut(4));

.to({alpha: 0, y: 75}, 500, createjs.Ease.getPowInOut(2))

.to({alpha: 0, y: 125}, 100)

.to({alpha: 1, y: 100}, 500, createjs.Ease.getPowInOut(2))

.to({x: 100}, 800, createjs.Ease.getPowInOut(2));


createjs.Tween.get(windowStage, {loop: false})

.to({x: 0, y: -1250}, 5000, createjs.Ease.getPowInOut(1));


createjs.Ticker.setFPS(60);

createjs.Ticker.addEventListener("tick", stage);

},


function tweenStop()

{

createjs.Tween.removeTweens(windowSprite);

},
