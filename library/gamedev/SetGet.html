Класс Set и Get

Наследование класса

/* Класс ячейки */

var Cell = function()

{

var graphics = new PIXI.Graphics();

graphics.lineStyle(2, 0xFF00FF, 1);

graphics.beginFill(0xFF00BB, 0.25);

graphics.drawRoundedRect(150, 450, 300, 100, 15);

graphics.endFill();


var cellType = "CELL_TYPE_CLEAR";

var that = {

getCellGraphics: function()

{

if(cellType == "CELL_TYPE_EMPTY") graphics = new PIXI.Graphics();

return graphics;

},

getCellType: function()

{

return cellType;

},

setCellType: function(value)

{

cellType = value;

}

};

return that;

};



Создание экземпляра класса и обращение к переменным


var cell = new Cell();

cell.setCellType("CELL_TYPE_EMPTY");

stage.addChild(cell.getCellGraphics());

console.log(cell.getCellType());
