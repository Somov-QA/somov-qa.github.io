События (Event)

var max = 60;

var min = 0;

var count = min;

var timer;


function startTimer()

{

       timer = setInterval(onComplete, 1000);

}


function onComplete()

{

       if(count === max)

       {

               count = min;

       } else {

               count++;

       }

}


function stopTimer()

{

       clearInterval(timer);

}
