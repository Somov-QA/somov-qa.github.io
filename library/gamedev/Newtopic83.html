Объединение объектов в группы

var game = new Phaser.Game(860, 730, Phaser.AUTO, '', { preload: preload, create: create, update: update });


function preload()

{

       game.load.image('background', './assets/background.png');

       game.load.image('ground', './assets/ground.png');

       game.load.image('ledge', './assets/ledge.png');

       game.load.image('star', './assets/star.png');

       game.load.spritesheet('man', './assets/man.png', 32, 48);

}


function create()

{

       game.add.sprite(0, 0, 'background');

}
