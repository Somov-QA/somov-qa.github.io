ПРИМЕЧАНИЕ: Перед удалением MovieClip нжно выключать stop()


var stage = new PIXI.Container();


function loadTexture()

{

var loader = new PIXI.loaders.Loader();

loader.add('flashAtlas',"./assets/image/flash.json");

loader.once('complete',onLoaded);

loader.load();

}


function onLoaded(loader, res)

{

var flash = loadAnimationTextures(12, 'flash_');

var animation = new PIXI.extras.MovieClip(flash);

animation.position.x = 40;

animation.position.y = 30;

animation.scale.x += 0.5;

animation.scale.y += 0.5;

animation.animationSpeed = 0.2;

animation.play();

stage.addChild(animation);

}


function loadAnimationTextures(countFrame, nameFrame)

{

var nameTexture;

var animTextures = [];

for(var i = 1; i < countFrame; i++)

{

if(i < 10)

{

nameTexture = nameFrame + '0' + i + '.png'

}else{

nameTexture = nameFrame + i + '.png'

}

var texture = PIXI.Texture.fromFrame(nameTexture);

animTextures.push(texture);

}

return animTextures;

}
