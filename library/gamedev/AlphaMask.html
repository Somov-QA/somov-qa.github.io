Маска (Alpha Mask текстура является маской)

var game;



function onInit()

{

       game = new Phaser.Game(800, 600, Phaser.CANVAS, 'phaser-example', {preload: preload, create: create, update: update, render: render });

}


window.addEventListener("load", onInit, false);


function preload()

{

game.load.image('cube', 'assets/texture.jpg');

game.load.image('mask', 'assets/mask_texture.png');

}


function create()

{

game.stage.backgroundColor = 0x4d4d4d;


game.add.text(64, 10, 'Source image', { font: '16px Arial', fill: '#ffffff' })

var cube = game.add.image(64, 32, 'cube');

cube.scale.setTo(0.3, 0.3);


game.add.text(400, 10, 'Alpha mask', { font: '16px Arial', fill: '#ffffff' })

var mask = game.add.image(400, 32, 'mask');

mask.scale.setTo(0.3, 0.3);


//  Создаём bitmap

var bmd = game.make.bitmapData(512, 512);


//  создаём альфа маску путом комбинирования двух текстур

bmd.alphaMask('cube', 'mask');


//  BitmapData это просто текстура.

//  Нужно применить маску к спрайту

game.add.image(game.world.centerX, 135, bmd).anchor.set(0.5, 0);

}

{

       

}

{
