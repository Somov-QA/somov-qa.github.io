var windowspriteSprite;


function windowspriteSpriteCreate(posX, posY, texture)

{

       windowspriteSprite = new PIXI.Sprite(texture);

       windowspriteSprite.position.x = posX;

       windowspriteSprite.position.y = posY;

       windowspriteSprite.scale.set(1.0);

       

       windowStage.addChild(windowspriteSprite);

       

       windowStartTween();

}


function windowStartTween()

{

       createjs.Tween.get(windowspriteSprite, {loop: false})

               .to({x: 400, y: 400}, 1000, createjs.Ease.getPowInOut(3))

               .call(onwindowTweenComplete); // событие выполнено

       createjs.Ticker.setFPS(60);        

}


function onwindowTweenComplete()

{

       windowStage.removeChild(this);

}
