var windowStyleText = {

   font : 'bold 12px Arial',

   fill : '#F7EDCA',

   stroke : '#500000',

   strokeThickness : 3,

   wordWrap : true,

   wordWrapWidth : 440

};


var windowButton;


function windowButtonCreate(buttonText, buttonTexture)

{

       var text = new PIXI.Text(buttonText, windowStyleText);

       text.x = 0;

       text.y = 0;

               

       windowButton = new PIXI.Sprite(buttonTexture);

       windowButton.name = "button";

       windowButton.position.x = 0;

       windowButton.position.y = 0;

       windowButton.interactive = true;

       windowButton.buttonMode = true;

       windowButton.tap = onWindowButtonClick;

       windowButton.click = onWindowButtonClick;

       windowButton.on('mousedown', onWindowButtonDown);

       windowButton.on('touchstart', onWindowButtonDown);

       windowButton.on('mouseup', onWindowButtonUp);

       windowButton.on('touchend', onWindowButtonUp);

       windowButton.on('mouseupoutside', onWindowButtonUp);

       windowButton.on('touchendoutside', onWindowButtonUp);


       windowButton.addChild(text);

       windowStage.addChild(windowButton);

}


function onWindowButtonDown()

{

       this.isdown = true;

       this.scale.set(0.95);

       this.position.x += 5;

}


function onWindowButtonUp()

{

       if(this.isdown)

       {

               this.isdown = false;

               this.scale.set(1.0);

               this.position.x -= 5;

       }

}
