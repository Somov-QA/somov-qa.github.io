Текстурирование и сегментация объекта

Движение по окружности

var cubeGeometry = new THREE.CubeGeometry(300, 300, 300, 7, 7, 7); // сегментация 7, 7, 7

var cubeTexture = new THREE.Texture();

var loader = new THREE.ImageLoader();


loader.load('assets/texture.jpg', function(event){

       cubeTexture.image = event;

       cubeTexture.needsUpdate = true;

});


var cubeMaterial = new THREE.MeshBasicMaterial({map: cubeTexture, overdraw: true});

var cube = new THREE.Mesh(cubeGeometry, cubeMaterial);

scene.add(cube);
