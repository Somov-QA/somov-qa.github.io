function preload()

{

   game.load.atlas('dice_1_atlas', 'assets/dice_1_atlas.png', 'assets/dice_1_atlas.json');

}



var dice1 = null;

var dice2 = null;

var tween;


if(dice1 !== null && dice1 !== null)

{

dice1.kill();

dice2.kill();

}


dice1 = game.add.button(150, 500, "dice_1_atlas", onButtonSelectDiceClick, this);

tween = game.add.tween(dice1);

tween.to({ x: (Math.random() / 0.1) * 10, y: (Math.random() / 0.1 ) * 25 }, 500, 'Linear');

tween.start();

dice1.animations.add('dice', Phaser.Animation.generateFrameNames('dice', 0, 10, '', 2), 15, false);

dice1.animations.play('dice');


dice2 = game.add.button(125, 500, "dice_1_atlas", onButtonSelectDiceClick, this);

tween = game.add.tween(dice2);

tween.to({ x: (Math.random() / 0.1) * 25, y: (Math.random() / 0.1 ) * 45 }, 500, 'Linear');

tween.start();

dice2.animations.add('dice', Phaser.Animation.generateFrameNames('dice', 0, 10, '', 2), 15, false);

dice2.animations.play('dice');



function onButtonSelectDiceClick (event)

{

   if (event.frame === 10) event.frame = 12;

   else event.frame = 10;

}
