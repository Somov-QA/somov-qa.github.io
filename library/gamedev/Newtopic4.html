Конструктор и прототип


function Class(value1, value2)        // определяем свойства экземпляра

{

       this.value1 = value1;

       this.value2 = value2;

this.arr = [0, 1, 2, 3];

}


Class.prototype = {                        // определяет методы и общие свойства

       constructor: Class,                // конструктор

       getValue: function()

       {

               return this.value1;

       }

};


var myclass = new Class("value", 200);

myclass.arr.push(4);

myclass.getValue();
