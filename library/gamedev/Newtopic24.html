function loadSounds()

{

       var queue = new createjs.LoadQueue();

       createjs.Sound.alternateExtensions = ["mp3"];

       queue.installPlugin(createjs.Sound);


       queue.on("progress", that.onSoundLoaderProcess);

       queue.on("complete", that.onSoundLoaderComplete);


       queue.loadFile({"id":"music1", "src":"assets/music/music1.mp3"});

queue = null;

}


function onSoundLoaderProcess(event)

{

       console.log(event.progress + " " + event.loaded + " " +event.total);

}

               

function onSoundLoaderComplete(event)

{

createjs.Sound.play("music1", {loop:-1});

}


function loadAssets()

{

var loader = new PIXI.loaders.Loader();

loader.add('bgImage','./assets/image/background.jpg');

loader.add('gameAtlas','./assets/image/quest/atlas/game.json');

loader.once('complete',onAssetsLoaded);

loader.once('progress',onProgressLoad);

loader.load();

}


function onProgressLoad()

{

console.log("Загрузка: " + (Match.round(this.progress)) + "%");

}


function onAssetsLoaded(loader, res)

{

var bgTexture = res.bgImage.texture;

}
