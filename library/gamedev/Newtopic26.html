var stage = new PIXI.Container();


var iconsAll = new Object();


function loadAssets()

{

var loader = new PIXI.loaders.Loader();

loader.add('gameAtlas','./assets/image/quest/atlas/game.json');

loader.once('complete',onAssetsLoaded);

loader.load();

}


function onAssetsLoaded(loader, res)

{

var iconTexture = PIXI.Texture.fromFrame('icon0.png');

iconsAll["icon0"] = iconBarakaTexture;

iconTexture = PIXI.Texture.fromFrame('icon1.png');

iconsAll["icon1"] = iconBarakaTexture;

showIcon();

}


function showIcon()

{

var icon;

for (var i = 0; i < iconsAll.length; i++)

{

icon = new PIXI.Sprite(iconsAll["icon" + i);

icon.position.x = 10;

icon.position.y = 50 + (30 * i);

icon.scale.set(0.85);

stage.addChild(icon);

}

}
