var game;


function onInit()

{

       game = new Phaser.Game(860, 730, Phaser.CANVAS, '', { preload: preload, create: create, update: update, render: render });

}


window.addEventListener("load", onInit, false);




function preload()

{

   game.load.atlas('dice_1_atlas', 'assets/dice_1_atlas.png', 'assets/dice_1_atlas.json');

}



function create()

{

   for(var i = 0; i < 7; i++)

   {      

       game.add.sprite(20 + (120 * i), 625, 'box');


       var button = game.add.button(40 + (120 * i), 650, 'dice_1_atlas', onButtonDiceClick, this, 0, 0, 0);

       button.animations.add('dice_1_atlas', Phaser.Animation.generateFrameNames('dice', 0, 10, '', 2), 15, true);

       button.onInputOver.add(onButtonDiceOver, this);

       button.onInputOut.add(onButtonDiceOut, this);

       button.onInputUp.add(onButtonDiceUp, this);


       button.frameName = "dice01";        // имя фрейма

       button.inputEnabled = true;        // флаг активности кнопки (разрешаем нажать)

   }

}


function onButtonDiceUp(event)

{

   

}


function onButtonDiceOver(event)

{

   

   event.animations.play('dice_1_atlas');

}


function onButtonDiceOut(event)

{

   event.animations.stop();

   if (event.frame === 10) event.frame = 12;

   else event.frame = 10;

}


function onButtonDiceClick (event)

{

   event.kill();

}
