Таймер

function Preloader()

{

       this.stage = new PIXI.Container();

}


Preloader.prototype = {

       constructor: Preloader,

       start: function()

       {

               var loader = new PIXI.loader.Loader();

               loader.add('preloaderTexture', "./assets/image.jpg");

               loader.once('complete', this.onComplete.bind(this));

               loader.load();

               loader = null;

       },

       onComplete: function(loader, res)

       {

               var sprite = new PIXI.Sprite(res.preloaderTexture.texture);

               this.stage.addChild(sprite)

               this.show();

               sprite = null;

       },

       show: function()

       {


       }

};
