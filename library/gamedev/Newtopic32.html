var stage = new PIXI.Container();


var menuStage;


var styleTextButton = {

font : 'bold 12px Arial',

fill : '#F7EDCA',

stroke : '#500000',

strokeThickness : 3,

wordWrap : true,

wordWrapWidth : 440

};


function menuShow()

{

var button;

var textButton;

var textButtonsRus = ["Кнопка 1","Кнопка 2","Кнопка 3"];


menuStage = new PIXI.Container();


for(var i = 0; i < textButtonsRus.length; i++)

{

textButton = new PIXI.Text(textButtonsRus[i], styleTextButton);

textButton.x = (170 / 2) - (textButton.width / 2);

textButton.y = 20;


button = new PIXI.Sprite(buttonTexture);

button.name = "button" + i;

button.position.x = (MAIN_WIDTH / 2) - (170 / 2);

button.position.y = (MAIN_HEIGH / 3) + (55 * i);

button.interactive = true;


/* события */

button.tap = onMenuButtonClick;

button.click = onMenuButtonClick;

button.on('mousedown', onMenuButtonDown);

button.on('touchstart', onMenuButtonDown);

button.on('mouseup', onMenuButtonUp);

button.on('touchend', onMenuButtonUp);

button.on('mouseupoutside', onMenuButtonUp);

button.on('touchendoutside', onMenuButtonUp);

button.addChild(textButton);


menuStage.addChild(button);

}

stage.addChild(menuStage);

console.log("Create window: menu");

}


function onMenuButtonDown()

{

this.isdown = true;

this.scale.set(0.95);

this.position.x += 5;

}


function onMenuButtonUp()

{

if(this.isdown)

{

this.isdown = false;

this.scale.set(1.0);

this.position.x -= 5;

}

}


function onMenuButtonClick()

{

if(this.name == "button0")

{

stage.removeChild(menuStage);

}

console.log("Menu click button: " + this.name);

}
