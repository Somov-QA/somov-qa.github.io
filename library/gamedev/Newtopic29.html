function initSprite(imageTexture)

{

var sprite = new PIXI.Sprite(imageTexture);

sprite.name = "Unit1";

sprite.position.x = 100;

sprite.position.y = 100;

sprite.interactive = true;

sprite.buttonMode = true;


// добавляем свои методы и свойства

sprite.unitType = "UNIT1";

sprite.flagRemove = false;

sprite.posColumnI = 0;

sprite.posRowJ = 0;


sprite.click = onUnitClick;

sprite.tap = onUnitClick;

}


function onUnitClick()

{

console.log("UNIT [Unit Click]["+this.name+"]: " + this.unitType + " | " + this.flagRemove);

}
