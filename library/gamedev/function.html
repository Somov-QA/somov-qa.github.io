Использование в виде класса


var Class(parent)

{

       var that = {

               name: "name",

               getName: function()

               {

                       return that.name;

               },

               setName: function(value)

               {

                       that.name = value;

               }

       };

       return that;

}


var class = Class(null);

class.setName("John");

console.log(class.getName());
