var GlobalLevel = function()                        // Класс уровня

{

       var that = {

               levelName : null,

           levelJson : null,

                   backgroundTexture : null

   };

       return that;

};


function loadAssets()

{

var loader = new PIXI.loaders.Loader();

loader.add('bdImage','./assets/image/background.jpg');

loader.add('level1','./assets/data/quest/levels/level1.json');

loader.once('complete',onLoaded);

loader.load();

}


function onLoaded(loader, res)

{

       var gLevel = new GlobalLevel();                        /* Класс уровня */

       gLevel.levelName = "Level1";

       gLevel.levelJson = res.level1;                        /* level1.json */

gLevel.backgroundTexture = res.bgImage.texture; /* background.jpg */

}
