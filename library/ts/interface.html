module Fabrique {

   export interface IDiceObject {

       id: number;

       type: boolean;

       star: boolean;

       enable?:boolean;

       remove?:boolean;

       values: number[];

       framesField: string[];

       framesActual: string[];

       indexFace:number;

       inField:boolean

   }

   

   export class Dice {

       

       private dices: any[] = [];

       

       constructor(){

                       

           for(let i:number = 0; i < numberDicesField; i++){


               let diceObj: IDiceObject = <IDiceObject>{};

               diceObj.id = i;

               diceObj.type = false;

               diceObj.star = false;

               diceObj.enable = false;

               diceObj.remove = false;

               diceObj.values = this.diceValues[indexValue][indexGroup];

               diceObj.framesField = this.diceFieldFrames[indexValue][indexGroup];

               diceObj.framesActual = this.diceActualFrames[indexValue][indexGroup];

               diceObj.indexFace = 0;

               diceObj.inField = true;

               this.dices.push(diceObj);

           }

       }

}
