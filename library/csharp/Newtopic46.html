- Определение классов.


class MyClass:BigClass // класс наследуется от другого класса BigClass

{

  public MyClass() // конструктор

  {

     _property = "";

  }

  public string GetProprty() // метод

  {

     return _property;

  }

  public string Property // свойство

  {

     get {return _property;}

     set {_property = value;}

  }

  private string _property; // поле

}
