Циклы

Синтаксис C# ››

- Цикл с предусловием

int w = 1;

while (w <= 5) {

w++;

}


- Цикл с постусловием

int d = 10;

do {

d--;

} while (d > 1);


- Цикл счётчик

for (int i = 0; i <= 10; i++) {

//...

}


- Цикл типа foreach выполняет обход набора элементов

int[] value = {2,4,3,5,1};

foreach (int count in value){

MessageBox.Show("Значение: " + value[count].ToString());

}


- Дополнительные операторы циклов

break; // завершение цикла

continue; //переход к выполнению следующей интерации

goto label; // переход к метке
