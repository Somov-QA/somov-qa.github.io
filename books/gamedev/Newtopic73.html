Цикл while


weekdays = {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"}


i = 1

while weekdays[i] do

   print(weekdays[i])

   i = i + 1

end


Цикл repeat until


i = 0

repeat

   i = i + 1

   print(weekdays[i])

until weekdays[i] == "Saturday"


Цикл for


-- от 1 до 10

for i = 1, 10 do

   print(i)

end


-- от 1 до 10 и увеличение на 2

for i = 1, 10, 2 do

   print(i)

end



-- в обратном направлении от 10 к 1

for i=10, 1, -1 do

   print(i)

end


t = { "a", "b", "c", "d" }

-- передрать последовательности

for i, v in ipairs(t) do

   print(v)

end
