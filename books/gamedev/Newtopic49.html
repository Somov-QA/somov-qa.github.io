var graphicOverlay = new Phaser.Graphics(this.game, 0 , 0);

graphicOverlay.beginFill(0x000000, 0.5);

graphicOverlay.drawRect(0,0, game.width, game.height);

graphicOverlay.endFill();

game.add.image(0,0,graphicOverlay.generateTexture()).inputEnabled = true;
