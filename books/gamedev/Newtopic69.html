Наполнение таблиц данными


table.insert - вставить значение

table.remove - удалить значение


self.field = {}


local tableColor = {

       row1 = {1,2,5,4,1,3,2},

       row2 = {1,3,4,3,3,5,3},

       row3 = {2,2,4,2,1,4,5},

       row4 = {3,3,1,1,5,4,3},

       row5 = {4,3,2,1,4,1,4},

       row6 = {5,5,3,3,2,3,1},

       row7 = {5,5,1,4,2,1,2}

}


for col = 1, 7 do

       for row = 1, 7 do

               

               table.insert(self.field, {

value = tableColor["row" .. row][col],

col = col,

row = row

})        

       end

end


print(self.field[table.getn(self.field)].value)

function init(self)

       self.fieldPos = {}


       local pos = vmath.vector3()

       for col = 1, 7 do

               for row = 1, 7 do

                       pos.x = 100 + (80 * col)

                       pos.y = 700 - (80 * row)

                       pos.z = 0.1

                       local colRow = "col" .. col .. ":row" .. row

                       self.fieldPos[colRow] = {x = pos.x, y = pos.y, z = pos.z}

               end

       end

end
