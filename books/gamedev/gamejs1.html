function Game(mainStage)

{

       this.mainStage = mainStage;

}


Game.prototype = {

       constructor: Game,


       config: {music:true, sound:true, MAIN_WIDTH:860, MAIN_HEIGH:730},

       assets: null,

       menu: null,


       onGame: function(event)

       {

               switch(event.detail.type)

               {

                       case 'menu_show':

                       {

                               this.menuShow();

                               break;

                       }


                       case 'menu_close':

                       {

                               this.menuClose();

                               break;

                       }


default:

               break;

               }

}


       menuShow: function()

{

this.menu = new Menu(this.assets);

this.menu.create();

this.mainStage.addChild(this.menu.show());


},


menuClose: function()

{

this.mainStage.removeChild(this.menu.windowStage);

this.menu = null;

}

};
