Взаимодействие объектов друг с другом (персонаж и звёзды)

var worldGroup;

var player;

var starsGroup;

var cursors;


function create()

{

       game.physics.startSystem(Phaser.Physics.ARCADE);


       // ---------------------------------------------


       game.physics.startSystem(Phaser.Physics.ARCADE);        // Используем аркадную физику

       worldGroup = game.add.group();

       worldGroup.enableBody = true;                                // Включаем физику для всех объектов группы


       var obj1 = worldGroup.create(0, game.world.heigh - 64, 'ground');

obj1.scale.setTo(2, 2);

obj1.body.immovable = true;                                        // включение тела объекта


var obj2 = worldGroup.create(400, 400, 'ledge');

obj2.body.immovable = true;                                        // включение тела объекта


var obj3 = worldGroup.create(-150, 250, 'ledge');

obj3.body.immovable = true;                                        // включение тела объекта


       // ---------------------------------------------


       player = game.add.sprite(32, game.world.height - 150, 'man');


game.physics.arcade.enable(player);                                // включаем физику персонажа


// свойства физики персонажа. (включаем небольшой отскок)

player.body.bounce.y = 0.2;                                        // отскок

player.body.gravity.y = 300;                                        // гравитация (сила притяжения)

player.body.collideWorldBounds = true;                        // сталкивание с границами мира


// две анимации: хотьба влево и вправо

player.animations.add('left', [0, 1, 2, 3], 10, true);

player.animations.add('right', [5, 6, 7, 8], 10, true);


       // ---------------------------------------------


starsGroup = game.add.group();                                // объединяем все звёзды в группу


starsGroup.enableBody = true;                                        // включаем физику для всех объектов группы


//  Создание звёзд и их равномерноя расстановка

for (var i = 0; i < 12; i++)

{

var star = starsGroup.create(i * 70, 0, 'star');        //  Создаём звезду внутри группы

star.body.gravity.y = 300;                                        //  добавляем гравитацию (силу притяжения)

star.body.bounce.y = 0.7 + Math.random() * 0.2;                //  отскок от поверхности

}


       // ---------------------------------------------


cursors = game.input.keyboard.createCursorKeys();        // Управление

}


function update()

{

       // проверка столкновений объектов(воспроизведение физики)

       game.physics.arcade.collide(player, worldGroup);        // персонаж

       game.physics.arcade.collide(starsGroup, worldGroup);        // звёзды


       // проверка перекрытия персонажа и звезд (для того чтобы удалить звезду)

       game.physics.arcade.overlap(player, starsGroup, removeStar, null, this);


       player.body.velocity.x = 0;                                        // остановка объекта


       // движение персонажа

       if (cursors.left.isDown)                                        // движение влево (нажата кнопка влево)

           {

               player.body.velocity.x = -150;

               player.animations.play('left');

           }

           else if (cursors.right.isDown)                                // движение вправо (нажата кнопка вправо)

           {

               player.body.velocity.x = 150;

               player.animations.play('right');

           }

           else                                                                        // останавливается

           {

               player.animations.stop();

               player.frame = 4;

           }


           //  прыжок (нажатие кнопки вверх)

           if (cursors.up.isDown && player.body.touching.down)

           {

               player.body.velocity.y = -350;

           }

}


function removeStar (player, star)

{

star.kill();                                                        // Удаляем звезду

}
