function loadAssets()

{

       var loader = new PIXI.loaders.Loader();

       loader.add('personages','./assets/data/personages.json');

       loader.on('progress', that.onAssetsLoaderProcess);

       loader.on('complete', that.onAssetsLoaderComplete);

       loader.load();

       loader = null;

}
