function Menu(assets)

{

       this.assets = assets;

}


Menu.prototype = {

       constructor: Menu,


       windowStage: null,

       backgroundSprite: null,


       create: function()

       {

               this.windowStage = new PIXI.Container();

               this.backgroundCreate();

       },


       backgroundCreate: function()

       {

               this.backgroundSprite = new PIXI.Sprite(this.assets.getAsset("backgroundTexture"));

               this.backgroundSprite.position.x = 0;

               this.backgroundSprite.position.y = 0;

               this.windowStage.addChild(this.backgroundSprite);

       },

       

       tweenStart: function()

       {

               createjs.Tween.get(this.backgroundSprite , {loop: true})

                       .to({x: -50, y: 0}, 2500, createjs.Ease.getPowInOut(3))

                       .to({x: 50, y: 0}, 2500, createjs.Ease.getPowInOut(3));

               createjs.Ticker.setFPS(60);

       },


       tweenStop: function()

       {

               createjs.Tween.removeTweens(this.backgroundSprite);

       },


       show: function()

       {

               this.tweenStart();

               return that.windowStage;

       },


       close: function()

       {

               this.tweenStop();

               for(var child in this.windowStage.children)

{

this.windowStage.removeChild(this.windowStage.children[child]);

               }


               window.dispatchEvent(new CustomEvent('game', {'detail': {'type':'menu_close', 'value':null}}));        

       }

};
