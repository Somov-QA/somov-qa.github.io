var enemiesAI = []; // Список

var Enemy = function() // Класс

{

var that = {

ai_name : null,

ai_hit_1 : 0,

ai_hit_2 : 0,

ai_hit_3 : 0,

ai_hit_4 : 0,

ai_hit_5 : 0,

ai_life : 0

};

return that;

};


function initEnemiesAI()

{

var fightersName = ["liukang", "kunglao", "johnnycage"];

var enemiesAI = [];

var enemy;

for (var i = 0; i < fightersName.length; i++)

{

enemy = new Enemy();

enemy.ai_name = fightersName[i];

enemy.ai_hit_1 = 10;

enemy.ai_hit_2 = 20;

enemy.ai_hit_3 = 30;

enemy.ai_hit_4 = 40;

enemy.ai_hit_5 = 50;

enemy.ai_life = 100;

enemiesAI.push(enemy);

console.log("[EnemiesAI]: " + enemiesAI[enemiesAI.length - 1].ai_name);

}

return enemiesAI;

}
