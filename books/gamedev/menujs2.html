function Menu(config)

{

      this.config = config;

}


Menu.prototype = {

      constructor: Menu,


      windowStage: "MENU",

     

      create: function()

      {

           console.log("МЕНЮ СОЗДАНО: + " + this.windowStage);

           this.show();  

      },


      show: function()

      {

             console.log("МЕНЮ КОНФИГУРАЦИЯ: + " + this.config);

             this.close();

      },


      close: function()

      {

             console.log("МЕНЮ ЗАКРЫТО: + " + this.windowStage);


window.dispatchEvent(new CustomEvent('game', {'detail': {'type':'menu_close', 'value':null}}));        

      }

};
