СЛУШАТЕЛЬ


window.addEventListener("game", onGame, false);


function onGame(event)

{

       switch(event.detail.type)

       {

               case 'onLoad':

               {

                       console.log(event.detail.value);

                       break;

               }

default:

       break;

       }

}


ВОЗВРАЩАЕМ СОБЫТИЕ СЛУШАТЕЛЮ


var Preloader = function()

{

       var that = {

               myText: "текст сообщения",

               myFunc: function()

               {

                       window.dispatchEvent(new CustomEvent('game', {'detail': {'type':'onLoad', 'value':that.myText}}));

               }

       };

       return that;

}
