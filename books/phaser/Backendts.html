class SolitaireBackend extends Fabrique.BackendGame implements NodeGame {

   public static GAME_ID: number = 3003;

   public static BONUS_COUNT: number = 1;

   public static NUMBER_OF_FIELD_COLS: number = 7;


   public gameover: boolean = false;

   public score: number = 0;

   public success: boolean = false;

   public shortGame: boolean = false;

   public bonusStars: number = 0;


   private call: string = 'init';

   private isPlaying: boolean = false;

   private gameDataObject: Fabrique.Deck;


   private field:Fabrique.ICardObject[][] = [];


   constructor(gameDataObject: Fabrique.Deck) {

       super();

       this.gameDataObject = gameDataObject;

       this.gameDataObject.shuffle();

   }


   public handleGameMove(command: any): void {

       this.call = command.type;


       this.success = false;

       if ('init' === command.type) {

           if (this.field.length === 0 && this.score === 0) {

               this.init();

           }

       } else if ('hit' === command.type) {

           this.hit();

       } else if ('combine' === command.type){

           this.combine(command.cards, command.originalColumn, command.movedColumn);

       }


       this.gameover = this.isGameOver();

       let response: any = this.generateResponse();

       response.call = command.type;

       response.success = this.success;


       if (command.type === 'init') {

           response.config = this.getConfig();

       }

       

       return response;

   }



   public generateResponse(): any {

       let response: any = {

           call: this.call,

           success: this.success,

           score: this.score,

           cardsindeck: this.gameDataObject.size(),

           gameover: this.gameover,

           isplaying: this.isPlaying,

           bonusStars: this.bonusStars,

       };


       return response;

   }


   private getConfig(): any {

       return {

           bonuscount: SolitaireBackend.BONUS_COUNT,

           numberoffieldcols: SolitaireBackend.NUMBER_OF_FIELD_COLS,

       };

   }


  private isGameOver(): boolean {

   }



   private init(): void {

       this.starID = this.getRandomID();

       this.score = 0;

       this.gameover = false;


       this.field = [];

       this.createField();

   }



   private createField():void {

       for(let i:number = 0; i < SolitaireBackend.NUMBER_OF_FIELD_COLS; i++){

           this.field.push([]);

       }


   }


   private hit():void {


   }    


   private combine(cards:Fabrique.ICardObject[], originalColumn:number, movedColumn: number):void {


   }




}
