module Fabrique {


   export class Card extends Phaser.Sprite {


       public cardObject:Fabrique.ICardObject;

       private spriteGroup:any;

       public endTweenSignal:Phaser.Signal;


       constructor(game:Phaser.Game, x:number, y:number, key:string, group:any, cardObject:Fabrique.ICardObject){

           super(game, x, y, key);


           this.game = game;

           this.cardObject = cardObject;

           this.spriteGroup = group;

           this.init();

       }


       private init():void{

           this.endTweenSignal = new Phaser.Signal();

           this.spriteGroup.add(this);

       }



       public addStar():void{

           var star:Phaser.Sprite = this.game.add.sprite(0, 0, Atlases.OtherArt, "bonusstar_pink.png");

           star.scale.set(0.8);

           this.addChild(star);

           star.x = this.width/2 - star.width/2;

           star.y = this.height/2 - star.height/2;

       }


   }

}
