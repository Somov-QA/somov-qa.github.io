       /**

        * Socket has been set up in the plash screen, and we have received the intial response. This data is passed to this state,

        * and we're gonna use that to fill the game

        *

        * @param initData

        */

       public init(initData: Quartz.Response): void {

           this.initData = initData;

           //We register the data event handler, each request to the server will have a response (except start) and that reponse will be handled

           Quartz.Socket.getInstance().on('data', (data: any) => this.handleGameMove(data));

           Quartz.Socket.getInstance().send(new Quartz.Request('start', Quartz.Session.getInstance()));

       }


       public create(): void {

           super.create();

           this.game.world.shutdown();



           var test:Phaser.Sprite = this.game.add.sprite(0, 0, 0, 0);


           this.game.events.onGameStart.dispatch();

           this.game.analytics.game.addEvent(new GA.Events.Design('Game:Start'));

           (<any>window).state = this;

           this.handleGameMove(this.initData);

           

           this.resize();

       }

       /**

        * All incoming server response are handled here. Response only come in after a request has been send.

        * @param data

        */

       public handleGameMove(data: Quartz.Response): void {

           this.data = data;


           // init

           if (data.get('call') === 'init') {

               this.updateField(data.get('field'));

               this.updateDeck(data.get('deck'));

           }


           // hit

           if (data.get('call') === 'hit') {

               this.updateDeck(data.get('deck'));

               this.updateDeckStack(data.get('deckStack'));

           }


           // combine (result success)

           if (data.get('call') === 'combine') {

               if(data.get('success') === true){

                   // !!! the card remains in the new location

               }else{

                   // !!! animation back card

               }

           }


           // getALL

           if (data.get('call') === 'getAll') {

               this.updateField(data.get('field'));

               this.updateDeck(data.get('deck'));

           }


           // getField

           if (data.get('call') === 'getField') {

               this.updateField(data.get('field'));

           }


           // getDeck

           if (data.get('call') === 'getDeck') {

               this.updateDeck(data.get('deck'));

           }


           // gameover

           if (data.get('gameover') === true) {

               this.goGameover(data);

           }

       }
