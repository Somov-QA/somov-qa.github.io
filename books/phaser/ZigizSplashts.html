module Fabrique {

   export interface FunnyGamesSplashConfig {

       nextState: string;

       link: string;

       preloadHandler: () => void;

       preloadTexts?: string[];

   }

   

   export class FunnyGamesSplash extends Phaser.State {

       public static Name: string = 'funnySplashScreen';


       public static IdealWidth: number = 463;


       public static MaxDots: number = 3;


       public name: string = FunnyGamesSplash.Name;


       private scaleFactor: number = 1;


       private created: boolean = false;


       private playButton: Phaser.Button;


       private background: SplashBackground;


       private config: FunnyGamesSplashConfig;


       private preloadText: Phaser.Text;


       private dataHandler: (data: Quartz.Response) => void;


       private preloadTexts: string[] = [

           'Fetching images',

           'Fetching more images',

           'Downloading sounds',

           'Calculating Pi'

       ];


       private dots: number = 1;


       private dotText: string = '.';


       constructor() {

           super();

       }


       /**

        * Get some data from the config, and check if the spine plugin was loaded

        *

        * @param config

        */

       public init(config: FunnyGamesSplashConfig, background: SplashBackground) {

           this.config = config;

           this.background = background;


           if (this.config.preloadTexts) {

               this.preloadTexts = this.config.preloadTexts;

           }


           this.game.scale.onSizeChange.add(this.resize, this);

       }


       /**

        * Create all the objects required for the splash screen

        */

       public preload(): void {

           this.game.add.existing(this.background);


           this.createObjects();


           this.scaleFactor = this.getScale();


           this.resizeObjects();


           this.startAnimation();


           this.game.load.onLoadStart.add(this.onPreloadStart, this);

           this.game.load.onFileComplete.add(this.onPreloadedFile, this)

           this.game.load.onLoadComplete.add(this.onPreloadEnd, this);

           

           this.config.preloadHandler();


           if (this.game.load.totalQueuedFiles() === 0) {

               this.onPreloadEnd();

           }

       }


       private t: number;

       private tId: number = 0;


       private onPreloadStart() {

           this.t = Date.now();

           this.preloadText.setText(this.preloadTexts[this.tId]);

       }


       private onPreloadedFile() {

           if (Date.now() - this.t > 2000) {

               this.t = Date.now();

               this.tId++;


               if (this.tId >= this.preloadTexts.length) {

                   this.tId = 0;

               }

           }


           if (this.dotText.length >= 3) {

               this.dotText = '.';

           } else {

               this.dotText += '.';

           }


           this.preloadText.setText(this.preloadTexts[this.tId] + this.dotText);

       }


       private onPreloadEnd() {

           //Set up the socket

           let GS: Quartz.Socket = Quartz.Socket.getInstance();

           GS.surpressErrorPopup();

           this.dataHandler = (data: Quartz.Response) => this.onData(data);

           GS.on('data', this.dataHandler);

           GS.connect('localhost', 5678, false);

           console.log('connecting!');

           //Do some audio decoding before continuing;

           Sounds.preloadList.forEach((assetName:string) =>{

               this.game.add.audio(assetName);

           });

           this.game.sound.setDecodedCallback(Sounds.preloadList, this.startGameAfterDecodingAudio, this);

       }


       private startGameAfterDecodingAudio(){

           this.preloadText.visible = false;

           if (!this.game.device.desktop) {

               this.game.input.onDown.add(this.playClick, this);

               this.animateButton();

           } else {

               this.game.time.events.add(Phaser.Timer.SECOND, () => {

                   console.log('sending init ',  Quartz.Socket.getInstance());

                   Quartz.Socket.getInstance().send(new Quartz.Request('init', Quartz.Session.getInstance()));

               });

           }

           

           //ToDo: uncomment block above and delete this before publish

           /*

           this.game.time.events.add(Phaser.Timer.SECOND, () => {

                   console.log('sending init ',  Quartz.Socket.getInstance());

                   Quartz.Socket.getInstance().send(new Quartz.Request('init', Quartz.Session.getInstance()));

           });

           */

       }


       public createObjects(): void {

           this.preloadText = this.game.add.text(this.game.width / 2, this.game.height / 2 + 230, '', {

               font: 'Arial',

               fontSize: 40,

               fill: '#ffffff',

               fontWeight: 'bold'

           });

           this.preloadText.anchor.set(0.5);


           // this.logo = this.game.add.image(0, 0, 'fgs_logo');

           // this.logo.anchor.set(0.5);

           // this.logo.inputEnabled = true;

           // this.logo.input.useHandCursor = true;

           // this.logo.events.onInputDown.add(() => {

           //     window.open(this.config.link, '_blank');

           // });

           // FunnyGamesSplash.IdealWidth = this.logo.getBounds().width * 1.5;


           this.playButton = this.game.add.button(this.game.width / 2, this.game.height / 2 + 230 * this.scaleFactor, 'fgs_play', this.playClick, this, 0, 0, 1, 0);

           this.playButton.anchor.set(0.5);

           this.playButton.scale.set(this.scaleFactor);

           this.playButton.visible = false;


           this.created = true;

       }


       public resizeObjects(): void {

           if (!this.created) {

               return;

           }

           

           this.preloadText.x = this.game.width / 2;

           this.preloadText.y = this.game.height / 2 + 230 * this.scaleFactor;

           this.preloadText.scale.set(this.scaleFactor);


           if (!this.game.device.desktop) {

               this.playButton.x = this.game.width / 2;

               this.playButton.y = this.game.height / 2;

           }

       }


       public playClick(): void {

           this.game.tweens.remove(this.buttonTween);

           var t = this.game.add.tween(this.playButton.scale);

           t.to({x: 0, y: 0}, 350, Phaser.Easing.Back.In).onComplete.add(() => {

               // this.game.state.start(this.config.nextState);

               Quartz.Socket.getInstance().send(new Quartz.Request('init', Quartz.Session.getInstance()));

           });

           t.start();

           this.game.input.onDown.remove(this.playClick, this);

       }



       /**

        * Data handler for the init state. Once we receive the init state we start the game

        *

        * @param data

        */

       private onData(data: Quartz.Response): void {

           if (data.get('call') === 'init') {

               this.game.state.start(this.config.nextState, false, false, data);

               Quartz.Socket.getInstance().off('data', this.dataHandler);

           }

       }


       public getScale(): number {

           if (!this.game.device.desktop) {

               if (this.game.width < this.game.height) {

                   return (this.game.width / 6 * 4) / FunnyGamesSplash.IdealWidth;

               } else {

                   return (this.game.width / 6 * 3) / FunnyGamesSplash.IdealWidth;

               }

           } else {

               if (this.game.width < FunnyGamesSplash.IdealWidth * 1.5) {

                   return (this.game.width / 6 * 4) / FunnyGamesSplash.IdealWidth;

               }

           }


           return 1;

       }


       public startAnimation(): void {

           if (!this.game.sound.touchLocked) {

               // this.game.sound.play('fg');

           }

       }


       private buttonTween: Phaser.Tween;


       private animateButton() {

           if (this.game.device.desktop) {

               return;

           }

           this.game.tweens.remove(this.buttonTween);


           this.playButton.scale.set(this.scaleFactor);

           this.playButton.visible = true;

           this.buttonTween = this.game.add.tween(this.playButton.scale);

           this.buttonTween.to({

               x: 1.14 * this.scaleFactor,

               y: 1.14 * this.scaleFactor

           }, 420, Phaser.Easing.Quadratic.InOut, true, 500);

           this.buttonTween.yoyo(true).onComplete.add(this.animateButton, this);

           this.buttonTween.start();

       }


       public update() {

           this.background.update();

       }


       public loadUpdate() {

           this.background.update();

       }


       public resize() {

           this.scaleFactor = this.getScale();

           this.background.resize(this.scaleFactor);

           this.resizeObjects();


           if (this.created) {

               this.animateButton();

           }

       }


       public shutdown() {

           this.game.scale.onSizeChange.remove(this.resize, this);

       }

   }

}
