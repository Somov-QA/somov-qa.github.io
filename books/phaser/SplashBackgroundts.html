module Fabrique {

   export class SplashBackground extends Phaser.Group {

       private triangles: Phaser.Polygon[] = [];


       private triangleSizes: number[] = [350, 650, 950, 1250];//220, 390, 560, 730];


       private back: Phaser.Graphics;


       private gradientBitmap: Phaser.BitmapData;


       private gradient: Phaser.Sprite;


       private t: Phaser.Graphics;


       constructor(game: Phaser.Game, scaleFactor: number) {

           super(game, null, 'splash-background');


           this.back = this.game.make.graphics(0, 0);

           this.t = this.game.make.graphics(0, 0);

           this.gradientBitmap = this.game.make.bitmapData(this.game.width, this.game.height);

           this.gradient = this.game.make.sprite(0, 0, this.gradientBitmap);


           this.add(this.back);

           this.add(this.t);

           this.add(this.gradient);


           this.resize(scaleFactor)

       }


       public resize(scaleFactor: number) {

           this.back.clear()

               .beginFill(0x33b0ff)

               .drawRect(0, 0, this.game.width, this.game.height);


           this.updateTrinagles(scaleFactor);


           this.t.x = this.game.width / 2;

           this.t.y = this.game.height / 2;

           this.t.clear();

           this.t.lineStyle(80 * scaleFactor, 0xffffff, 0.3);

           this.triangles.forEach((triangle: Phaser.Polygon) => {

               this.t.drawPolygon(triangle);

           });


           this.t.angle = 30;


           this.gradientBitmap.resize(this.game.width, this.game.height);

           var grd = this.gradientBitmap.context.createLinearGradient(0,0,0,this.game.height - 200);

           grd.addColorStop(0, "rgba(51, 176, 255, 0)");

           grd.addColorStop(1, "rgba(51, 176, 255, 1)" );

           this.gradientBitmap.context.fillStyle=grd;

           this.gradientBitmap.context.clearRect(0, 0, this.game.width, this.game.height);

           this.gradientBitmap.context.fillRect(0,0,this.game.width, this.game.height);

       }



       private updateTrinagles(scaleFactor: number): void {

           var center: Phaser.Point = new Phaser.Point(this.game.width / 2, this.game.height / 2);

           this.triangles = [];

           //create polygons

           this.triangleSizes.forEach((radius:number, i: number) => {

               radius *= scaleFactor;

               this.triangles.push(new Phaser.Polygon([

                   new Phaser.Point(radius * Math.cos(0), radius * Math.sin(0)),

                   new Phaser.Point(radius * Math.cos((1./3)*(2*Math.PI)), radius * Math.sin((1./3)*(2*Math.PI))),

                   new Phaser.Point(radius * Math.cos((2./3)*(2*Math.PI)), radius * Math.sin((2./3)*(2*Math.PI))),

                   new Phaser.Point(radius * Math.cos(0), radius * Math.sin(0)),

               ]));

           });

       }


       public update() {

           this.t.angle += 0.3;

       }

   }

}
