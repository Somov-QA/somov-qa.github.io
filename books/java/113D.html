11. Выбор диапазона 3D анимации

Практика libGDX ››

В данном разделе мы рассмотрим два способа выбора участка анимации из совокупности анемаций.

1) Получаем объект в формате FBX и конвертируем его как описано в главе 8. Конвертирование FBX в G3DB/G3DJ

2) Скопируем в папку assets наш объект, файл объекта *.g3dj и файл текстуры *.png или другом формате



3) В папке testAnim3D создаём новый класс TestSelectAnim3D интерфейс которого Screen



4) Класс TestSelectAnim3d в конструкторе создаёт камеру, свет, привязываем событие вращения камерой.

C помощью G3dModelLoader читается файл объекта g3dj (формат json) и создаём экземпляр ModelInstance.

Чтобы воспроизвести анимацию нужно использовать AnimationController.

Запуск анимации по прежнему осуществляется методом setAnimation() но на этот раз мы указываем диапазон анимации.


controller.setAnimation("Take 001", 5.0f, 1.0f, 3, 1.0f,new AnimationController.AnimationListener(){ });


где первый параметр - это идентификатор;

второй параметр - это точка начала анимации;

третий параметр - это продолжительность (или лучше сказать длина) анимации начиная с начальной точки;

четвертый параметр - это количество повторов анимации;

пятый параметр - это скорость анимации;

шестой параметр - это обработчик событий.


В нашем случае в обработчике событий мы перехватываем завершение анимации для того чтобы изменить участок следующей анимации и запустить её снова.


@Override

public void onEnd(AnimationController.AnimationDesc animation) { }


Вот таким образом мы меняем начальную точку анимации, длину анимации и количество повторов анимации.


animation.offset = 13.3f;

animation.duration = 0.56f;

animation.loopCount = 5;


Полный код примера:


package com.mygdx.game.testAnim3D;

import com.badlogic.gdx.Files;
import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.GL20;
import com.badlogic.gdx.graphics.PerspectiveCamera;
import com.badlogic.gdx.graphics.g3d.Environment;
import com.badlogic.gdx.graphics.g3d.Model;
import com.badlogic.gdx.graphics.g3d.ModelBatch;
import com.badlogic.gdx.graphics.g3d.ModelInstance;
import com.badlogic.gdx.graphics.g3d.attributes.ColorAttribute;
import com.badlogic.gdx.graphics.g3d.loader.G3dModelLoader;
import com.badlogic.gdx.graphics.g3d.utils.AnimationController;
import com.badlogic.gdx.graphics.g3d.utils.CameraInputController;
import com.badlogic.gdx.utils.JsonReader;
import com.mygdx.game.MyGdxGame;

/**
* Created by Cartish on 12.10.2016.
*/

public class TestSelectAnim3D implements Screen {
   final MyGdxGame game;
   private PerspectiveCamera camera;
   private ModelBatch modelBatch;
   private Model model;
   private ModelInstance modelInstance;
   private Environment environment;
   private AnimationController controller;

   public CameraInputController camController;
   public Integer index = 0;

   public TestSelectAnim3D(final MyGdxGame myGdxGame) {
       game = myGdxGame;

       camera = new PerspectiveCamera(75, Gdx.graphics.getWidth(), Gdx.graphics.getHeight());
       camera.position.set(0f,0f,200f);
       camera.lookAt(0f,0f,0f);
       camera.near = 0.1f;
       camera.far = 3000.0f;
       camera.update();

       camController = new CameraInputController(camera);
       Gdx.input.setInputProcessor(camController);

       modelBatch = new ModelBatch();

       // D3DJ ------------------------------------------------------
       G3dModelLoader modelLoader = new G3dModelLoader(new JsonReader());
       model = modelLoader.loadModel(Gdx.files.getFileHandle("selectAnim3D/npc_all_animations_v01.g3dj", Files.FileType.Internal));
       modelInstance = new ModelInstance(model);
       modelInstance.transform.setTranslation(0.0f, -100.0f, 0.0f);
       // ------------------------------------------------------------

       environment = new Environment();
       environment.set(new ColorAttribute(ColorAttribute.AmbientLight, 0.8f, 0.8f, 0.8f, 1.0f));

       controller = new AnimationController(modelInstance);

       controller.setAnimation("Take 001", 5.0f, 1.0f, 3, 1.0f,new AnimationController.AnimationListener(){
           @Override
           public void onEnd(AnimationController.AnimationDesc animation) {
               if(index == 0) {
                   animation.offset = 13.3f;
                   animation.duration = 0.56f;
                   animation.loopCount = 5;
                   index = 1;
               }else
               if(index == 1) {
                   animation.offset = 8.0f;
                   animation.duration = 2.0f;
                   animation.loopCount = 2;
                   index = 2;
               }else
               if(index == 2){
                   animation.offset = 5.0f;
                   animation.duration = 1.0f;
                   animation.loopCount = 3;
                   index = 0;
               }
           }

           @Override
           public void onLoop(AnimationController.AnimationDesc animation) { }
       });
   }

   @Override
   public void show() {    }

   @Override
   public void render(float delta) {
       Gdx.gl.glViewport(0, 0, Gdx.graphics.getWidth(), Gdx.graphics.getHeight());
       Gdx.gl.glClearColor(1, 1, 1, 1);
       Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT | GL20.GL_DEPTH_BUFFER_BIT);

       camera.update();

       controller.update(Gdx.graphics.getDeltaTime());

       modelBatch.begin(camera);
       modelBatch.render(modelInstance, environment);
       modelBatch.end();
   }

   @Override
   public void resize(int width, int height) {    }

   @Override
   public void pause() {    }

   @Override
   public void resume() {    }

   @Override
   public void hide() {    }

   @Override
   public void dispose() {
       modelBatch.dispose();
       model.dispose();
   }
}


Если всё было сделано правильно в результате вы увидите выборочную анимацию нпс.
