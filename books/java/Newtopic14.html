Объекты

Point p;                        // объявление объекта (p равно null)

p = new Point(10,10);        // создание объекта (выдиление помяти)

p.x = 20;                // доступ к свойствам объекта

p.show();                // вызов методов

p = null;                        // очистка объекта (сборщик мусора)
