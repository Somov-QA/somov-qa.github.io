interface IApple {

       String red = "red";

       String yellow = "yellow";

String green = "green";

}


interface ITangerine {

       String orange = "red";

}


interface IFruits extends IApple, ITangerine  {

       String fruit1 = "Apple";

String fruit2 = "Tangerine";

void setFruit (int number1, int number2);

String getFruit();

}


class MyFruits implements IFruits {

       String name;

       

       public String getFruit() {

               return "Fruit: " + this.name;

}


public void setFruit (int number1, int number2){

       switch(number1){

               case 0:

                       this.name = this.fruit1;

break;

               case 1:

                       this.name = this.fruit2;

break;

       default:

               this.name = this.fruit1;

               break;

}

}


public static void main (String args[]){

       MyFruits mf1 = new MyFruits();

       MyFruits.setFruit(0, 2);

       System.out.println(MyFruits.getFruit());

}
