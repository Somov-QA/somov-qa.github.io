1) Создаём папку testColoring создаём новый класс TestColoring интерфейс которого Screen



2) Для работы с примитивами необходимо использовать Pixmap, Texture, Sprite.

Инициализация и создание примитивов квадрат и круг

Визуализация примитивов описывается в функции render с помощью batch


@Override
public void render(float delta) {
   Gdx.gl.glClearColor(0, 0, 0.2f, 1);
   Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT);

   camera.update();

   batch.setProjectionMatrix(camera.combined);
   batch.begin();
   batch.draw(sprite1, sprite1.getX(), (480-sprite1.getHeight())- sprite1.getY());
   batch.draw(sprite2, sprite2.getX(), (480-sprite2.getHeight())- sprite2.getY());
   batch.end();
}


Чтобы раскрасить примитив можно использовать не сколько способов:

Раскрасить примитив "Квадрат" с помощью другого примитива "Круг". Для этого нужно получить данные текстуры из спрайта и внести в них изменения (нарисовав круг в том месте где был щелчок мыши), а затем вернуть изменённую текстуру текстуру обратно в спрайт.


public boolean mouseMoved (int x, int y) {
   if(sprite != null) {
       float posX = x - sprite.getX();
       float posY = y - sprite.getY();
       System.out.println("Image Clicked X=" + posX + " Y=" + posY);
       texture = sprite.getTexture();
       pixmap = texture.getTextureData().consumePixmap();
       if(pixmap.getPixel((int)posX, (int)posY) == -1) {
           pixmap.setColor(Color.VIOLET);
           pixmap.fillCircle((int) posX, (int) posY, 5);

           texture = new Texture(pixmap);
           sprite.setTexture(texture);
       }
   }
   return true;
}

Полный код примера:


package com.mygdx.game.testColoring;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.GL20;
import com.badlogic.gdx.graphics.OrthographicCamera;
import com.badlogic.gdx.graphics.Pixmap;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.Sprite;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.InputAdapter;
import com.mygdx.game.MyGdxGame;

/**
* Created by Cartish on 25.10.2016.
*/

public class TestColoring implements Screen {
   final MyGdxGame game;
   OrthographicCamera camera;
   SpriteBatch batch;

   Pixmap pixmap;
   Texture texture;
   Sprite sprite;
   Sprite sprite1;
   Sprite sprite2;

   public TestColoring(final MyGdxGame myGdxGame) {
       game = myGdxGame;
       batch = new SpriteBatch();
       camera = new OrthographicCamera();
       camera.setToOrtho(false, 800, 480);

       createGraphicObjects();
   }

   private void createGraphicObjects(){
       pixmap = new Pixmap(64, 64, Pixmap.Format.RGBA8888 );
       pixmap.setColor( 1, 1, 1, 1 );
       pixmap.fillRectangle(0,0,64,64);
       texture = new Texture(pixmap);
       sprite1 = new Sprite(texture);
       sprite1.setBounds(100,100,64,64);

       pixmap = new Pixmap(64, 64, Pixmap.Format.RGBA8888 );
       pixmap.setColor( 1, 1, 1, 1 );
       pixmap.fillCircle(32, 32, 25);
       texture = new Texture(pixmap);
       sprite2 = new Sprite(texture);
       sprite2.setBounds(300,100,64,64);

       sprite = null;
       Gdx.input.setInputProcessor(new InputAdapter(){
           @Override
           public boolean touchDown(int screenX, int screenY, int pointer, int button) {
               if(sprite1.getBoundingRectangle().contains(screenX, screenY)) {
                   sprite = sprite1;
               }
               if(sprite2.getBoundingRectangle().contains(screenX, screenY)) {
                   sprite = null;
                   float posX = screenX - sprite2.getX();
                   float posY = screenY - sprite2.getY();
                   texture = sprite2.getTexture();
                   pixmap = texture.getTextureData().consumePixmap();
                   pixmap.setColor(Color.RED);
                   for(int i = 0; i < 25; i++){
                       for(int j = 0; j < 25; j++){
                           if(pixmap.getPixel((int)posX+i, (int)posY+j) == -1){
                               pixmap.drawPixel((int)posX+i, (int)posY+j);
                               texture = new Texture(pixmap);
                               sprite2.setTexture(texture);
                           }
                       }
                   }
                   //sprite = sprite2;
               }
               return true;
           }

           @Override
           public boolean touchUp (int x, int y, int pointer, int button) {
               return true;
           }

           public boolean mouseMoved (int x, int y) {
               if(sprite != null) {
                   float posX = x - sprite.getX();
                   float posY = y - sprite.getY();
                   System.out.println("Image Clicked X=" + posX + " Y=" + posY);
                   texture = sprite.getTexture();
                   pixmap = texture.getTextureData().consumePixmap();
                   if(pixmap.getPixel((int)posX, (int)posY) == -1) {
                       pixmap.setColor(Color.VIOLET);
                       pixmap.fillCircle((int) posX, (int) posY, 5);

                       texture = new Texture(pixmap);
                       sprite.setTexture(texture);
                   }
               }
               return true;
           }

       });
   }

   @Override
   public void show() {   }

   @Override
   public void render(float delta) {
       Gdx.gl.glClearColor(0, 0, 0.2f, 1);
       Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT);

       camera.update();

       batch.setProjectionMatrix(camera.combined);
       batch.begin();
       batch.draw(sprite1, sprite1.getX(), (480-sprite1.getHeight())- sprite1.getY());
       batch.draw(sprite2, sprite2.getX(), (480-sprite2.getHeight())- sprite2.getY());
       batch.end();
   }

   @Override
   public void resize(int width, int height) {   }

   @Override
   public void pause() {   }

   @Override
   public void resume() {   }

   @Override
   public void hide() {   }

   @Override
   public void dispose() {
       batch.dispose();
       texture.dispose();
       pixmap.dispose();
   }
}
